פַּרְעֹה, הַתַּנִּים הַגָּדוֹל, מֶלֶךְ מִצְרַיִם – הַמַּעֲצָמָה הַגְּדוֹלָה בְּיוֹתֵר בָּעוֹלָם,
רָצָה לְהַשְׁמִיד אֶת עַם יִשְׂרָאֵל בְּצוּרָה מְתֻחְכֶּמֶת עַל יְדֵי שֶׁמִּנָּה עֲלֵיהֶם שָׂרֵי מִסִּים לְעַנּוֹתָם.

אֵין הוּא מְפַחֵד לֹא מֵהַשֵּׁם וְלֹא מִמֹּשֶׁה עַבְדּוֹ… הוּא אוֹמֵר: "מִי הַשֵּׁם אֲשֶׁר אֶשְׁמַע בְּקוֹלוֹ"?…

וְלֹא עוֹד אֶלָּא שֶׁהוּא מִתְפָּאֵר וְאוֹמֵר: "לִי יְאוֹרִי וַאֲנִי עֲשִׂיתִינִי".

גַּם הַמַּכּוֹת שֶׁהֻנְחֲתוּ עָלָיו וְעַל  עַמּוֹ לֹא הִרְתִּיעוּ אוֹתוֹ,

וְאַף שֶׁרָאָה אוֹתוֹת וּמוֹפְתִים הִתְיַחֵס בַּאֲדִישׁוּת כְּאִלּוּ לֹא קָרָה דָּבָר.

גַּם כַּאֲשֶׁר עֲבָדָיו אָמְרוּ: "אֶצְבַּע אֱלֹקִים הִוא",

וַאֲפִלּוּ שֶׁהִתְחַנְּנוּ אֵלָיו: "שַׁלַּח אֶת הָאֲנָשִׁים וְיַעַבְדוּ אֶת אֱלֹקֵיהֶם",

פַּרְעֹה לֹא שִׁנָּה אֶת עֶמְדוֹתָיו וְלֹא נִכְנַע.

הַקָּדוֹשׁ בָּרוּךְ הוּא שׁוֹאֵל: "עַד מָתַי מֵאַנְתָּ לֵעָנֹת מִפָּנָי"?…

מָה עוֹד צָרִיךְ פַּרְעֹה לְקַבֵּל כְּדֵי לְהִכָּנַע וְלִהְיוֹת עָנִי וּשְׁפַל רוּחַ?…
רַק כַּאֲשֶׁר הַמַּכּוֹת מִתְעַצְּמוֹת וּמַגִּיעוֹת לְשִׂיאָן בְּמַכַּת בְּכוֹרוֹת וּבְמִצְרַיִם נִשְׁמַעַת צְעָקָה גְּדוֹלָה,
אֲפִלּוּ פַּרְעֹה מֶלֶךְ מִצְרַיִם נִבְהַל וְשֶׁלֹּא כְּדַרְכּוֹ כְּמֶלֶךְ הוּא קָם בַּחֲצוֹת לַיִל,

רָץ בְּגוּפוֹ וּמְחַזֵּר עַל בָּתֵּי עֲבָדָיו וּמַעֲמִידָם… וּמְחַזֵּר עַל פִּתְחֵי הָעִיר, וְצוֹעֵק:

הֵיכָן מֹשֶׁה שָׁרוּי?… הֵיכָן אַהֲרֹן שָׁרוּי?…

וְכַאֲשֶׁר הוּא פּוֹגֵשׁ אֶת מֹשֶׁה הוּא מַרְכִּין רֹאשׁוֹ וּמִתְחַנֵּן:

"קוּמוּ צְאוּ מִתּוֹךְ עַמִּי גַּם אַתֶּם גַּם בְּנֵי יִשְׂרָאֵל וּלְכוּ עִבְדוּ אֶת הַשֵּׁם כְּדַבֶּרְכֶם"…

יֵשׁ לוֹ רַק בַּקָּשָׁה אַחַת: "וּבֵרַכְתֶּם גַּם אוֹתִי".
פַּרְעֹה הָרָשָׁע, בְּמַהוּתוֹ הוּא הַשֹּׁרֶשׁ וְהַמָּקוֹר לַקְּלִפּוֹת הַמַּסְתִּירוֹת אֶת הָאֱלֹקוּת.
פַּרְעֹה הוּא סֵמֶל הַגַּאֲוָה וְהַיֵּשׁוּת, כִּי שָׁרְשׁוֹ שֶׁל פַּרְעֹה נַעֲלֶה וְגָבוֹהַּ מְאוֹד.

כָּל דָּבָר בָּעוֹלָם יֵשׁ לוֹ שֹׁרֶשׁ בַּקְּדֻשָּׁה הָעֶלְיוֹנָה וְשָׁרְשׁוֹ בַּקְּדֻשָּׁה הוּא מַהוּתוֹ הָאֲמִתִּית.

רַק כְּשֶׁיּוֹרֵד לְמַטָּה פְּעָמִים הוּא מִתְהַפֵּךְ לְטֻמְאָה,

וְכָל הַגָּבוֹהַּ יוֹתֵר יוֹרֵד לְמַטָּה יוֹתֵר, כְּמוֹ פַּרְעֹה שֶׁהָפַךְ לְשִׂיא קְלִפַּת הַיֵּשׁוּת וְהַגַּאֲוָה,

וּכְדֵי לְהַכְנִיעַ אֶת פַּרְעֹה צָרִיךְ לְהַגִּיעַ לְשָׁרְשׁוֹ.

אָכֵן שֹׁרֶשׁ פַּרְעֹה הַמֻּשְׁרָשׁ בְּשֹׁרֶשׁ עֶלְיוֹן אוֹרוֹ גָּדוֹל… עַד שֶׁבְּ"בֵית פַּרְעֹה" יֵשׁ אוֹרוֹת בְּלִי גְּבוּל,

אֲפִלּוּ מֹשֶׁה רַבֵּנוּ, רַבָּן שֶׁל נְבִיאִים, כֵּלָיו מוּעָטִים מִלְּהָכִיל אֶת הָאוֹרוֹת שֶׁל פַּרְעֹה,

לָכֵן הוּא חוֹשֵׁשׁ לְהִכָּנֵס לִפְנִימִיּוּת פַּרְעֹה (לְבֵית פַּרְעֹה) וּלְקַבֵּל אֶת מַדְרֵגַת פַּרְעֹה שֶׁבַּקְּדֻשָּׁה.
רַק הַקָּדוֹשׁ בָּרוּךְ הוּא הַנּוֹשֵׂא הֲפָכִים וְנִמְנָע הַנִּמְנָעוֹת יָכוֹל לְהַצִּיעַ לְמֹשֶׁה:
"בֹּא אֶל פַּרְעֹה" – בֹּא נֵלֵךְ יַחַד לְפַרְעֹה.

וְהַקָּדוֹשׁ בָּרוּךְ הוּא מוֹלִיךְ אֶת מֹשֶׁה לִפְנִימִיּוּת פַּרְעֹה,

חֶדֶר לִפְנִים מֵחֶדֶר כְּדֵי לִכְבֹּשׁ אֶת פַּרְעֹה וּלְבַטֵּל אֶת תֹּקֶף גַּאֲוָתוֹ מִכָּל וְכָל.

תֹּקֶף כָּזֶה שֶׁאֵין לוֹ שׁוּם תַּקָּנָה אֶלָּא רַק עַל שְׁבִירָה.

מֻכְרָחִים לִשְׁבֹּר אֶת תֹּקֶף גַּאֲוָתוֹ בְּשָׁרְשָׁהּ,

יֵשׁ לִפְגֹּעַ בְּעִקָּר מְצִיאוּתוֹ – בְּתוֹכִיּוּתוֹ וּבִפְנִימִיּוּתוֹ,

וּמִמֵּילָא כָּל הַדְּבָרִים הַטְּפֵלִים מִתְבַּטְּלִים.

וְאָכֵן מֹשֶׁה, נִבְרָא מֻגְבָּל, מִתְאַחֵד עִם הַקָּדוֹשׁ בָּרוּךְ לִשְׁבִירַת קְלִפַּת פַּרְעֹה בְּשָׁרְשָׁהּ.

כָּךְ יָכוֹל עַם יִשְׂרָאֵל לַעֲמֹד הָכֵן לִקְרַאת הַגְּאֻלָּה מִמִּצְרַיִם שֶׁתַּכְלִיתָהּ מַתַּן תּוֹרָה שֶׁלְּמַעְלָה מֵהַגְבָּלָה.

וּמִכֵּיוָן שֶׁכָּל יְהוּדִי הוּא "עוֹלָם קָטָן" שֶׁבִּשְׁבִילוֹ נִבְרָא הָעוֹלָם.

וְכָל עִנְיְנֵי הָעוֹלָם תְּלוּיִים וְנַעֲשִׂים בְּלִבּוֹ תְּחִלָּה –

הֲרֵי שֶׁכָּל יְהוּדִי צָרִיךְ לְקַיֵּם: "כָּל מַעֲשֶׂיךָ יִהְיוּ לְשֵׁם שָׁמַיִם"… וּ"בְכָל דְּרָכֶיךָ דָּעֵהוּ"…

אֶלָּא שֶׁבְּכָל אֶחָד יֵשׁ קְלִפּוֹת "מִצְרַיִם" וּ"בָבֶל" שֶׁמְּצֵרִים לוֹ וּמְבַלְבְּלִים אוֹתוֹ.

וְהַקָּדוֹשׁ בָּרוּךְ הוּא הָאוֹהֵב אֶת יִשְׂרָאֵל עַמּוֹ – "כִּי נַעַר יִשְׂרָאֵל וָאֹהֲבֵהוּ",

בָּא בִּכְבוֹדוֹ וּבְעַצְמוֹ ("הוּא וְלֹא אַחֵר") לְסַיֵּעַ לָנוּ לְבַטֵּל אֶת תֹּקֶף הַקְּלִפָּה.

"שֶׁאִלְמָלֵא הַקָּדוֹשׁ בָּרוּךְ הוּא עוֹזְרוֹ מִי יָכוֹל לוֹ?"…

הוּא פּוֹנֶה לְנִיצוֹץ מֹשֶׁה רַבֵּנוּ שֶׁבְּכָל יְהוּדִי וְאוֹמֵר לוֹ "בֹּא אֶל פַּרְעֹה" –

הָבָה נִכָּנֵס יַחְדָּו לְבֵית פַּרְעֹה, שִׂיא הַיֵּשׁוּת וְהַגַּאֲוָה, נִלָּחֵם בּוֹ, וּנְבַטֵּל אוֹתוֹ,

וְאָז מִמֵּילָא יִתְבַּטְּלוּ כָּל הַדְּבָרִים הַלֹּא רְצוּיִים הַמִּתְלַוִּים לָזֶה.

בְּדֶרֶךְ זוֹ אָנוּ מְקַיְּמִים אֶת מִצְוַת יְצִיאַת מִצְרַיִם בְּרוּחָנִיּוּת תָּמִיד בְּכָל עֵת וּבְכָל שָׁעָה,

וְעַל יְדֵי הַיְּצִיאָה מִמִּצְרַיִם "עֶרְוַת הָאָרֶץ",

אָנוּ הוֹפְכִים "אַרְצִיּוּת" לְ"אֶרֶץ שֶׁרָצְתָה לַעֲשׂוֹת רְצוֹן קוֹנָהּ",

וְיוֹרְשִׁים אֶת אֶרֶץ יִשְׂרָאֵל – "אֶרֶץ טוֹבָה וּרְחָבָה", "אֶרֶץ אֲשֶׁר תָּמִיד עֵינֵי הַשֵּׁם אֱלֹקֶיךָ בָּהּ".

וּבְקָרוֹב מַמָּשׁ נִכָּנֵס לְ"פַּלְטֵרִין שֶׁל מֶלֶךְ" – חֶדֶר לְפָנִים מֵחֶדֶר,

לִירוּשָׁלַיִם עִיר הַקֹּדֶשׁ, וּמִשָּׁם לְבֵית הַמִּקְדָּשׁ וּמִשָּׁם לְקֹדֶשׁ הַקָּדָשִׁים

בַּגְּאֻלָּה הָאֲמִתִּית וְהַשְּׁלֵמָה עַל יְדֵי הָרַבִּי מְשִׁיחַ צִדְקֵנוּ.
