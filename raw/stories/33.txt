בִּגְלַל שֶׁאַתֶּם מְאֹד יְקָרִים לָנוּ אֲנַחְנוּ רוֹצִים לוֹמַר לָכֶם מַשֶּׁהוּ מְאֹד חָשׁוּב. אַתֶּם צְרִיכִים לָדַעַת, שֶׁאֶחָד הַדְּבָרִים הֲכִי מְסֻכָּנִים שֶׁיֵּשׁ בָּעוֹלָם הוּא: הָאֵשׁ. הָאֵשׁ מְאֹד מְסֻכֶּנֶת כִּי הִיא יְכוֹלָה לִשְׂרֹף הַכֹּל! הִיא לֹא שָׂמָה לֵב אִם זֶה מַשֶּׁהוּ רַע אוֹ טוֹב, הִיא פָּשׁוּט שׂוֹרֶפֶת הַכֹּל! הִיא יְכוֹלָה לִשְׂרֹף אֲפִילוּ סֵפֶר תּוֹרָה!! לִפְנֵי הַרְבֵּה שָׁנִים, כְּשֶׁה' בָּרָא אֶת הָאֵשׁ הוּא בָּרָא אוֹתָהּ כְּדֵי שֶׁנּוּכַל לְהִשְׁתַּמֵּשׁ בָּהּ לְבַשֵּׁל, וּלְהִתְחַמֵּם.. אֲבָל מִי שֶׁלֹּא נִזְהָר יָכוֹל חָלִילָה לְהִשָּׂרֵף. וְלָכֵן, אֲנַחְנוּ מְבַקְּשִׁים מִכֶּם שֶׁ-לְּ-ע-וֹ-לָ-ם לֹא תַדְלִיקוּ אֵשׁ, אֶלָּא רַק אִם אֲנַחְנוּ הַהוֹרִים נִמְצָאִים אִתְּכֶם, אוֹ מִישֶׁהוּ גָדוֹל אַחֵר. בְּסֵדֶר?.. וַאֲנַחְנוּ רוֹצִים שֶׁאַתֶּם תִּהְיוּ הָאַחֲרָאִים עַל הַיְּלָדִים הַקְּטַנִּים שֶׁלִּפְעָמִים מוֹצְאִים גַּפְרוּרִים אוֹ מַצִּית וְרוֹצִים לְהַדְלִיק, אִם תִּרְאוּ דָבָר כָּזֶה, תַּעַצְרוּ אוֹתָם וְתָרוּצוּ אֵלֵינוּ מִיָּד וְתַגִּידוּ לָנוּ, בְּסֵדֶר? אֶפְשָׁר לִסְמֹךְ עֲלֵיכֶם? כָּל הַכָּבוֹד.. וְכָעֵת, נַתְחִיל אֶת הַסִּפּוּר הַיָּפֶה שֶׁכַּמּוּבָן קָשׁוּר לָאֵשׁ.. אָז תִּהְיוּ בְּשֶׁקֶט.. שְׁשְׁשְׁ...

לִפְנֵי שָׁנִים רַבּוֹת הָיָה מֶלֶךְ עֲרָבִי שֶׁרָצָה לִלְמֹד אֶת סֵפֶר תַּנַ"ךְ. וַאֲפִילוּ שֶׁהַתַּנַ"ךְ הוּא הַסֵּפֶר שֶׁל עַם יִשְׂרָאֵל וְהַמֶּלֶךְ הָיָה גוֹי, בְּכָל זֹאת זֶה עִנְיֵן אוֹתוֹ מְאֹד וְהוּא הִתְחִיל לִלְמֹד בּוֹ. הוּא קָרָא.. וְקָרָא.. וְקָרָא.. הוּא קָרָא עַל אַבְרָהָם אָבִינוּ, עַל יִצְחָק וְיַעֲקֹב, עַל יְצִיאַת מִצְרַיִם וּמַעֲמַד הַר סִינַי, וְאַחֲרֵי שֶׁהוּא סִיֵּם אֶת חֲמֵשֶׁת חוּמְשֵׁי תּוֹרָה הוּא הִמְשִׁיךְ לִקְרֹא אֶת הַנְּבִיאִים, עַל יְהוֹשֻׁעַ בִּן נוּן, עַל שְׁמוּאֵל הַנָּבִיא, דָּוִד הַמֶּלֶךְ וּשְׁלֹמֹה הַמֶּלֶךְ.. וְכָךְ הוּא קָרָא עוֹד וָעוֹד.. עַד שֶׁהוּא הִגִּיעַ לַסִּפּוּר הַיָּפֶה שֶׁסִּפַּרְנוּ לָכֶם אֶתְמוֹל, אַתֶּם זוֹכְרִים אֶת הַסִּפּוּר? נָכוֹן, עַל חֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה שֶׁנִּצְּלוּ מֵהָאֵשׁ.. וְאָז פָּתַח הַמֶּלֶךְ הָעֲרָבִי אֶת שְׁתֵּי הָעֵינַיִם שֶׁלּוֹ וְאָמַר: "מָה? הַאִם יָכוֹל לִהְיוֹת שֶׁיִּקְרֶה כָּזֶה נֵס שֶׁשְּׁלֹשָׁה אֲנָשִׁים נִכְנְסוּ לְתוֹךְ הָאֵשׁ וְלֹא יִקְרֵה לָהֶם כְּלוּם??.. אֲנִי חַיָּב לִרְאוֹת אֶת זֶה בָּעֵינַיִם שֶׁלִּי..." וְאָז עָלָה לוֹ רַעֲיוֹן נוֹרָא, הוּא אָמַר: "הִנֵּה, יֵשׁ בַּמַּלְכוּת שֶׁלִּי כַּמָּה וְכַמָּה יְהוּדִים, אֲנִי אֶבְחַר שְׁלוֹשָׁה חֲכָמִים מֵהֶם וְאֶזְרֹק אוֹתָם לָאֵשׁ וְאֶרְאֶה אִם הֵם לֹא יִשָּׂרְפוּ.."

מִיַּד שָׁלַח הַמֶּלֶךְ שְׁלִיחִים לְהָבִיא אֵלָיו כַּמָּה חֲכָמִים מֵעַם יִשְׂרָאֵל. כְּשֶׁהֵם הִגִּיעוּ הֵם שָׁאֲלוּ: "כֵּן, אֲדוֹנִי הַמֶּלֶךְ. מָה רָצִיתָ מֵאִתָּנוּ?" וְהַמֶּלֶךְ הָעֲרָבִי אָמַר: "קָרָאתִי בַּתּוֹרָה שֶׁלָּכֶם שֶׁחֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה נִזְרְקוּ לָאֵשׁ וְלֹא קָרָה לָהֶם כְּלוּם, לָכֵן, אֲנִי רוֹצֶה לְנַסּוֹת לִזְרֹק גַּם אֶתְכֶם לָאֵשׁ, כִּי גַם אַתֶּם חֲכָמִים.." הַחֲכָמִים פָּחֲדוּ וְאָמְרוּ לוֹ: "אֲדוֹנִי הַמֶּלֶךְ, חֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה הָיוּ צַדִּיקִים גְּדוֹלִים מְאֹד, וְלָכֵן קָרָה לָהֶם נֵס כָּזֶה, וְאִלּוּ אֲנַחְנוּ לֹא צַדִּיקִים כְּמוֹתָם.." אֲבָל כַּמָּה שֶׁהֵם בָּכוּ וְהִסְבִּירוּ לַמֶּלֶךְ הָעֲרָבִי, הוּא לֹא רָצָה לְוַתֵּר לָהֶם. אֶלָּא הוּא אָמַר בְּקוֹל רָם: "אֲנִי כְּבָר גָּזַרְתִּי אֶת הַגְּזֵרָה הַזּוֹ, וּמַה שֶׁהַמֶּלֶךְ גּוֹזֵר - צְרִיכִים לְקַיֵּם, וְלָכֵן אֲנִי הוֹלֵךְ לִזְרֹק אֶתְכֶם לָאֵשׁ וְשׁוּם דָּבָר לֹא יַעֲזֹר לָכֶם!!" הַחֲכָמִים רָאוּ שֶׁאֵין לָהֶם בְּרֵירָה, הֵם בִּקְּשׁוּ מִמֶּנּוּ: "לְפָחוֹת, תִּתֵּן לָנוּ שְׁלוֹשִׁים יוֹם שֶׁנִּתְכּוֹנֵן וְנַחֲזֹר בִּתְשׁוּבָה.." וְהַמֶּלֶךְ הִסְכִּים.

הֵם הָלְכוּ לַבַּיִת וְחָשְׁבוּ: "מָה נַעֲשֶׂה? הַמֶּלֶךְ הַזֶּה עַקְשָׁן גָּדוֹל.. הוּא יִזְרֹק אוֹתָנוּ לָאֵשׁ.. הֵם הֶחְלִיטוּ לַעֲשׂוֹת צוֹם שְׁלוֹשָׁה יָמִים עַד שֶׁיַּעֲנוּ לָהֶם מֵהַשָּׁמַיִם מָה לַעֲשׂוֹת.. הֵם לֹא אָכְלוּ וְלֹא שָׁתוּ, רַק יָשְׁבוּ בְּבֵית הַכְּנֶסֶת וְהִתְפַּלְּלוּ עִם דְּמָעוֹת. יְלָדִים יְקָרִים, אַתֶּם צְרִיכִים לָדַעַת שֶׁמִּי שֶׁמִּתְפַּלֵּל עִם דְּמָעוֹת, הַתְּפִלָּה שֶׁלּוֹ בְּוַדַּאי מִתְקַבֶּלֶת אֵצֶל בּוֹרֵא עוֹלָם.

הַחֲכָמִים סִכְּמוּ בֵּינֵיהֶם שֶׁכָּל מִי שֶׁיַּחֲלֹם חֲלוֹם בַּלַּיְלָה, יָבֹא בַּבֹּקֶר וִיסַפֵּר אֶת הַחֲלוֹם לַחֲבֵרִים. בֹּקֶר אֶחָד, הִתְעוֹרֵר אֶחָד הַחֲכָמִים וְאָמַר לַחֲבֵרָיו: "חָלַמְתִּי חֲלוֹם, וּבַחֲלוֹם אָמְרוּ לִי פָּסוּק אֶחָד שֶׁיֵּשׁ בּוֹ פַּעֲמַיִים אֶת הַמִּילָה "כִּי" וְשָׁלוֹשׁ פְּעָמִים אֶת הַמִּילָה "לֹא" וַאֲנִי לֹא זוֹכֵר מַהוּ הַפָּסוּק.. " אָמְרוּ לוֹ הַחֲכָמִים הָאֲחֵרִים: זֶה פָּסוּק מְאֹד מְיֻחָד שֶׁכָּתוּב בַּסֵּפֶר יְשַׁעְיָהוּ, וְשָׁם כָּתוּב: "כִּי תֵלֵךְ בְּמוֹ אֵשׁ -לֹא תִכָּוֶה" כְּלוֹמַר, שֶׁהָאֵשׁ לֹא תִשְׂרֹף אוֹתְךָ. וְזֶה סִימָן שֶׁאַתָּה תִזְכֶּה לְהִנָּצֵל מֵהָאֵשׁ. אוֹתוֹ חָכָם, הָלַךְ לַמֶּלֶךְ הָעֲרָבִי וְאָמַר לוֹ: "אֲנִי מוּכָן שֶׁתִּזְרֹק אוֹתִי לְתוֹךְ הָאֵשׁ.."

הַחַיָּלִים שֶׁל הַמֶּלֶךְ קָשְׁרוּ לוֹ אֶת הַיָּדַיִם וְאֶת הָרָגְלַיִם וְהֵבִיאוּ אוֹתוֹ לַמְּדוּרָה שֶׁהֵכִינוּ, זוֹ הָיְתָה מְדוּרָה גְדוֹלָה מְאֹד מְאֹד. וְהַחַיָּלִים זָרְקוּ אוֹתוֹ לָאֵשׁ הַבּוֹעֶרֶת.. "בּוּם!" וּמָה קָרָה? לְפֶתַע, הָאֵשׁ שֶׁבָּעֲרָה בַּמְּדוּרָה הִתְחַלְּקָה לִשְׁלוֹשָׁה שְׁבִילִים, וּבְתוֹךְ הַשְּׁבִילִים הִגִּיעוּ שְׁלוֹשָׁה צַדִּיקִים לְקַבֵּל אֶת פָּנָיו. כָּל אֶחָד מֵהַצַּדִּיקִים הָאֵלּוּ אָמַר כַּמָּה פְּסוּקִים מְיֻחָדִים, וְלֶחָכָם לֹא קָרָה כְּלוּם.. הַמֶּלֶךְ הָעֲרָבִי רָאָה אֶת זֶה וְכִמְעַט הִתְעַלֵּף.. הוּא בִּקֵּשׁ לְהוֹצִיא אוֹתוֹ מֵהָאֵשׁ וְעָשָׂה לוֹ כָּבוֹד גָּדוֹל.

יְלָדִים יְקָרִים, אַתֶּם רוֹצִים לִרְאוֹת אֶת הַפְּסוּקִים שֶׁאוֹתָם צַדִּיקִים אָמְרוּ?  בְּסֵדֶר גָּמוּר, נְגַלֶּה לָכֶם שֶׁאֶת הַפְּסוּקִים הָאֵלּוּ אוֹמְרִים בְּכָל יוֹם שֵׁנִי וַחֲמִישִׁי בְּבֵית הַכְּנֶסֶת. זֶה נִקְרָא "תַּחֲנוּן אָרוֹךְ שֶׁל שֵׁנִי וַחֲמִישִׁי" וְזֶה נִמְצָא בַּסִּדּוּר בִּתְפִלַּת שַׁחֲרִית וּמַתְחִיל בַּמִּלִּים: "וְהוּא רַחוּם". וּבִגְלַל הַסִּפּוּר הַזֶּה שֶׁמְּגַלֶּה לָנוּ שֶׁאֵלּוּ פְּסוּקִים מְאֹד קְדוֹשִׁים וּמְיֻחָדִים, אֲנַחְנוּ אוֹמְרִים אוֹתָם כָּל יוֹם שֵׁנִי וַחֲמִישִׁי, וּצְרִיכִים לוֹמַר אֶת זֶה דַוְקָא בַּעֲמִידָה וְלֹא בִּישִׁיבָה.. וְכָךְ ה' שׁוֹמֵעַ אֶת הַתְּפִלָּה שֶׁלָּנוּ וְשׁוֹמֵר גַּם עָלֵינוּ.

זֶהוּ חֲמוּדִים, הַסִּפּוּר הִסְתַּיֵּם, אֲבָל אַתֶּם לֹא שׁוֹכְחִים מָה הַתַּפְקִיד שֶׁנָּתַנּוּ לָכֶם, נָכוֹן? אַתֶּם זוֹכְרִים מַהוּ? נָכוֹן, לְהַשְׁגִּיחַ עַל הַיְלָדִים הַקְּטַנִּים שֶׁלֹּא יַדְלִיקוּ גַּפְרוּרִים וְלָבוֹא לְסַפֵּר לָנוּ מִיָּד.
