תַּגִּידוּ, הַאִם אַתֶּם יוֹדְעִים מָה זֶה 'מַצַּב רוּחַ'? לְכָל אָדָם יֵשׁ זְמַנִּים שֶׁבָּהֶם הוּא שָׂמֵחַ וְיֵשׁ זְמַנִּים שֶׁבָּהֶם הוּא עָצוּב. כְּשֶׁקּוֹרִים לוֹ דְבָרִים שְׂמֵחִים - הוּא נִהְיָה שָׂמֵחַ, וּכְשֶׁקּוֹרִים לוֹ דְבָרִים עֲצוּבִים - הוּא נִהְיָה עָצוּב. אֲבָל, אַתֶּם צְרִיכִים לָדַעַת, שֶׁלְּעַם יִשְׂרָאֵל יֵשׁ תְּרוּפָה מְיֻחֶדֶת נֶגֶד הָעַצְבוּת, כֵּן, כְּמוֹ שֶׁיֵּשׁ תְּרוּפָה נֶגֶד כְּאֵבֵי רֹאשׁ, וּתְרוּפָה לִכְאֵב גַּב.. כָּךְ יֵשׁ תְּרוּפָה לְעַצְבוּת! אֶת הַתְּרוּפָה הַזּוֹ יֵשׁ רַק לְעַם יִשְׂרָאֵל. אַתֶּם יוֹדְעִים מַהִי הַתְּרוּפָה?.. אַתֶּם רוֹצִים קְצָת מֵהַתְּרוּפָה הַזּוֹ?.. טוֹב, אָז נְגַלֶּה לָכֶם שֶׁהַתְּרוּפָה נֶגֶד הָעַצְבוּת הִיא: אֱמוּנָה! מִי שֶׁמַּאֲמִין בַּה' וְיוֹדֵעַ שֶׁהַכֹּל לְטוֹבָה - הוּא לֹא נִהְיָה עָצוּב. וְהַיּוֹם, נְסַפֵּר לָכֶם סִפּוּר עַל אֶחָד הַצַּדִּיקִים שֶׁהִשְׁתַּמֵּשׁ בִּתְרוּפָה זוֹ. אַתֶּם מוּכָנִים? יֹפִי, אֲנַחְנוּ מַתְחִילִים.

לִפְנֵי שָׁנִים רַבּוֹת, חַי בְּחוּץ לָאָרֶץ הַצַּדִּיק הַגָּדוֹל רַבִּי יִשְׂרָאֵל בַּעַל שֵׁם טוֹב. מִדֵּי פַּעַם, הָיָה הַבַּעַל שֵׁם טוֹב מְבַקֵּשׁ מִתַּלְמִידָיו לַעֲלוֹת לָעֲגָלָה הַמְיֻחֶדֶת שֶׁהָיְתָה לוֹ. בְּמָה הִיא הָיְתָה מְיֻחֶדֶת? שֶׁהֵם הָיוּ יוֹשְׁבִים בָּעֲגָלָה וְהַסּוּסִים הָיוּ דוֹהֲרִים לְבַד עַד הַמָּקוֹם שֶׁהָיוּ צְרִיכִים לְהַגִּיעַ.. פַּעַם אַחַת, הוּא עָלָה עִם הַתַּלְמִידִים שֶׁלּוֹ עַל הָעֲגָלָה, וְהַסּוּסִים הִתְחִילוּ לִדְהֹר, הֵם דָּהֲרוּ וְדָהֲרוּ בְּלִי הַפְסָקָה שָׁעוֹת אֲרֻכּוֹת. בֵּינְתַיִם, הִגִּיעַ כְּבָר יוֹם שִׁשִּׁי וְהַסּוּסִים עֲדַיִן דּוֹהֲרִים.. הַבַּעַל שֵׁם טוֹב הִתְחִיל לִדְאֹג כִּי עוֹד מְעַט הַשַּׁבָּת הַקְּדוֹשָׁה נִכְנֶסֶת וּמָה יִהְיֶה? הוּא הִתְחִיל לִהְיוֹת קְצָת עָצוּב.. אֲבָל מִיָּד הוּא הִשְׁתַּמֵּשׁ בַּתְּרוּפָה שֶׁאָמַרְנוּ. אַתֶּם זוֹכְרִים מַהִי הַתְּרוּפָה? נָכוֹן, אֱמוּנָה! הוּא הֶאֱמִין שֶׁהַכֹּל מֵה' וְהִתְחַזֵּק בְּשִׂמְחָה. לְפֶתַע הַסּוּסִים עָצְרוּ. בָּרוּךְ ה', סוֹף סוֹף אֶפְשָׁר לָרֶדֶת מֵהָעֲגָלָה.. כָּל הַתַּלְמִידִים יָרְדוּ וְרָאוּ שֶׁהֵם נִמְצָאִים בְּאֶמְצַע הַיַּעַר, מָה עוֹשִׂים? הֵם הִתְחִילוּ לִצְעֹד וּלְחַפֵּשׂ אֵיזֶה בַּיִת.. פִּתְאוֹם אָמַר אֶחָד הַתַּלְמִידִים: "הֵי, אֲנִי רוֹאֶה אֵיזֶה בַּיִת לְיָדֵנוּ.." הֵם צָעֲדוּ לְשָׁם וּבָרוּךְ ה' הָיְתָה שָׁם מְזוּזָה. הֵם הֵבִינוּ שֶׁגָּרִים שָׁם יְהוּדִים וְחָשְׁבוּ שֶׁשָּׁם יוּכְלוּ לַעֲשׂוֹת אֶת הַשַּׁבָּת בְּצוּרָה טוֹבָה.



הֵם דָּפְקוּ בַּדֶּלֶת, הַדֶּלֶת נִפְתְּחָה וּבַפֶּתַח עָמַד אִישׁ גָּבוֹהַּ, עִם שְׂעָרוֹת פְּרוּעוֹת, וְהַבְּגָדִים שֶׁלּוֹ הָיוּ כְּמוֹ שֶׁל בִּרְיוֹן. הוּא הִסְתַּכֵּל עֲלֵיהֶם בְּפָנִים כּוֹעֲסוֹת וְשָׁאַל בְּקוֹל עָבֶה : "מָה אַתֶּם רוֹצִים??" וְהַתַּלְמִידִים אָמְרוּ: "אֲנַחְנוּ רוֹצִים לַעֲשׂוֹת אֶצְלְךָ אֶת הַשַּׁבָּת.." הָאִישׁ אָמַר לָהֶם: "שְׁטֻיּוֹת!! אֲנִי לֹא צָרִיךְ אֲנָשִׁים כְּמוֹתְכֶם אֶצְלִי בְּשַׁבָּת.. תַּמְשִׁיכוּ בַּדֶּרֶךְ שֶׁלָּכֶם.." הַתַּלְמִידִים נִבְהֲלוּ וְאָמְרוּ: "לֹא, בְּבַקָּשָׁה.. אֲנַחְנוּ מִתְחַנְּנִים אֵלֶיךָ.. עוֹד מְעַט הַשְּׁקִיעָה.. הַשַּׁבָּת נִכְנֶסֶת וַאֲנַחְנוּ לֹא רוֹצִים לְחַלֵּל אוֹתָהּ.." הוּא הִסְתַּכֵּל עֲלֵיהֶם וְאָמַר: "אֲנִי מוּכָן לְהַכְנִיס אֶתְכֶם אֲבָל עִם כַּמָּה תְנָאִים", "מַה הֵם הַתְּנָאִים?" שָׁאֲלוּ הַתַּלְמִידִים, וְהוּא אָמַר: "אַתֶּם צְרִיכִים לְהִתְפַּלֵּל בְּשֶׁקֶט, וּלְהִתְפַּלֵּל מַהֵר מַהֵר, וְלֹא לִבְדּוֹק אֶת הָאֹכֶל שֶׁאֲנִי נוֹתֵן לָכֶם אִם הוּא כָּשֵׁר כְּמוֹ שֶׁאַתֶּם מַקְפִּידִים.. אִם אַתֶּם מוּכָנִים לַתְּנָאִים שֶׁלִּי - אַכְנִיס אֶתְכֶם לַבַּיִת שֶׁלִּי.."

הַתַּלְמִידִים הָיוּ מְאֹד עֲצוּבִים, כִּי הֵם רְגִילִים בְּדֶרֶךְ כְּלָל לְהִתְפַּלֵּל בְּשַׁבָּת בְּקוֹל רָם וּבְשִׂמְחָה, וְלַעֲבֹד אֶת ה' בְּכָל הַכֹּחַ. אֲבָל הַפַּעַם לֹא הָיְתָה לָהֶם בְּרֵירָה וְהֵם הִסְכִּימוּ. הָאִישׁ הֶרְאָה לָהֶם אֵיפֹה לְהִכָּנֵס וְהֵם חִכּוּ שָׁם בְּשֶׁקֶט.. לְאַחַר כַּמָּה דַקּוֹת, נִגַּשׁ הַבַּעַל שֵׁם טוֹב לָאִישׁ וְאָמַר לוֹ: "סְלִיחָה אֲדוֹנִי, אוּלַי יֵשׁ כָּאן אֵיזֶה נָהָר שֶׁאוּכַל לִטְבֹּל בּוֹ לִכְבוֹד שַׁבָּת?" אֵיךְ שֶׁהָאִישׁ שָׁמַע אֶת הַבַּעַל שֵׁם טוֹב הוּא נִהְיָה עַצְבָּנִי וְהִתְחִיל לִצְעֹק: "מָה?? מִקְוֶה?? הִזְהַרְתִּי אֶתְכֶם שֶׁלֹּא תַתְחִילוּ עִם כָּל הַשְּׁטוּיוֹת שֶׁלָּכֶם!! עַכְשָׁו, מַהֵר כֻּלָּם לָצֵאת הַחוּצָה מֵהַבַּיִת שֶׁלִּי.. אַתֶּם תִּשְׁנוּ בָּרְחוֹב!!!" הַבַּעַל שֵׁם טוֹב וְהַתַּלְמִידִים אָמְרוּ: "טוֹב, סְלִיחָה.. סְלִיחָה.. זֶה הָיָה בְּלִי כַּוָּנָה.. אֲנַחְנוּ לֹא נְבַקֵּשׁ יוֹתֵר שׁוּם דָּבָר.." מֵאוֹתוֹ הָרֶגַע הַבַּעַל שֵׁם טוֹב וְהַתַּלְמִידִים יָשְׁבוּ בְּשֶׁקֶט, הֵם הִתְפַּלְּלוּ בְּלַחַשׁ וְיָשְׁבוּ לַסְּעוּדָה. הָאִישׁ הֵבִיא קְעָרָה אַחַת שֶׁל אֹכֶל, שָׂם אוֹתָהּ בְּאֶמְצַע הַשֻּׁלְחָן וְכֻלָּם הָיוּ צְרִיכִים לֶאֱכוֹל מֵאוֹתָהּ קְעָרָה, וְגַם הָאִישׁ הַזֶּה אָכַל מֵאוֹתָהּ קְעָרָה וְזֶה לֹא הָיָה נָעִים בִּכְלָל.. אַחַר כָּךְ הֵם יָשְׁנוּ עַל הָרִצְפָּה בְּלִי שְׂמִיכוֹת וּבְלִי כָּרִית וְהָיָה לָהֶם קַר מְאֹד, בַּבֹּקֶר הֵם גַּם אָכְלוּ אֹכֶל לֹא טָעִים.. וְכָךְ כָּל הַשַּׁבָּת הֵם הָיוּ עֲצוּבִים נוֹרָא.. אֲבָל הַבַּעַל שֵׁם טוֹב הִתְחַזֵּק בֶּאֱמוּנָה וְחִכָּה לְהָבִין לָמָּה זֶה קָרָה?.. וְהִנֵּה, הִגִּיעַ מוֹצָאֵי שַׁבָּת.

לִפְנֵי שֶׁהַבַּעַל שֵׁם טוֹב עָמַד לָצֵאת מֵהַבַּיִת יַחַד עִם הַתַּלְמִידִים, הִגִּיעַ אֲלֵיהֶם בַּעַל הַבַּיִת אֲבָל הַפַּעַם הוּא חִיֵּךְ חִיּוּךְ גָּדוֹל.. וְהַשְּׂעָרוֹת שֶׁלּוֹ הָיוּ מְסֻדָּרוֹת.. "הֵי, מָה קָרָה לוֹ?" לְפֶתַע, נִפְתְּחָה הַדֶּלֶת וּבַפֶּתַח עָמְדָה אִשָּׁה. הַתַּלְמִידִים לֹא הֵבִינוּ מֵאֵיפֹה הִיא צָצָה? הֲרֵי כָּל הַשַּׁבָּת לֹא הָיְתָה כָּאן אִשָּׁה?.. הִיא נִגְּשָׁה לַבַּעַל שֵׁם טוֹב וְאָמְרָה לוֹ: "שָׁלוֹם עָלֶיךָ, רַבִּי!" הַבַּעַשֵׁ"ט הִתְפַּלֵּא וְאָמַר: "מֵאֵיפֹה אַתְּ יוֹדַעַת שֶׁאֲנִי רַבִּי? וְאִם יָדַעְתְּ לָמָּה נָתַתְּ לִי לִסְבֹּל כָּל הַשַּׁבָּת?" וְאָז הָאִשָּׁה אָמְרָה לוֹ כָּךְ: "כְּבוֹד הָרַב, קוֹרְאִים לִי שָׂרָה וַאֲנִי הָיִיתִי הַמְּשָׁרֶתֶת אֶצְלְךָ בַּבַּיִת.." הַבַּעַשֵׁ"ט הִתְחִיל לְהִזָּכֵר וְהִיא אָמְרָה: "אֲנִי הָיִיתִי יְתוֹמָה וּמִסְכֵּנָה, וְאִשְׁתְּךָ הָיְתָה מְטַפֶּלֶת בִּי כְּמוֹ בַּת. פַּעַם הִיא חָפְפָה לִי אֶת הָרֹאשׁ וְסֵרְקָה אוֹתִי חָזָק וְזֶה כָּאַב לִי. אֲנִי צָעַקְתִּי בְּקוֹל וְאִשְׁתְּךָ נָתְנָה לִי סְטִירָה.. אַתָּה רָאִיתָ אֶת זֶה וְלֹא אָמַרְתָּ לָהּ כְּלוּם.. וּבִגְלַל זֶה, בַּשָּׁמַיִם הָיָה עָלֶיךָ כַּעַס גָּדוֹל וְרָצוּ שֶׁאַתָּה תְאַבֵּד אֶת הָעוֹלָם הַבָּא שֶׁלְּךָ בִּגְלַל זֶה.. בַּעֲלִי שֶׁהוּא צַדִּיק נִסְתָּר, הִתְפַּלֵּל עָלֶיךָ הַרְבֵּה וּבַסּוֹף הִסְכִּימוּ בַּשָּׁמַיִם שֶׁבִּמְקוֹם שֶׁתְּאַבֵּד אֶת הָעוֹלָם הַבָּא, תִסְבֹּל שַׁבָּת אַחַת שֶׁהִיא מֵעֵין עוֹלָם הַבָּא וְכָךְ יִתְכַּפֵּר לְךָ.. וְלָכֵן, סְבַלְתֶּם כָּל הַשַּׁבָּת. אֲבָל עַכְשָׁו הַכֹּל בְּסֵדֶר.." הַבַּעַשֵׁ"ט שָׂמַח נוֹרָא וְרָאָה אֵיךְ שֶׁהַכֹּל לְטוֹבָה, כַּמּוּבָן שֶׁהֵם לֹא נָסְעוּ מִיָד אֶלָּא נִשְׁאֲרוּ עִם בַּעֲלָה הַצַּדִּיק וְלָמְדוּ תוֹרָה כָּל הַלַּיְלָה.

יְלָדִים יְקָרִים, לָמַדְנוּ הַיּוֹם שְׁנֵי דְּבָרִים חֲשׁוּבִים. דָּבָר רִאשׁוֹן: לֹא לִפְגֹעַ בְּאַף יְהוּדִי. וְדָבָר שֵׁנִי, תָּמִיד לְהִתְחַזֵּק בֶּאֱמוּנָה וְלֹא לִהְיוֹת עֲצוּבִים, כִּי הַכֹּל לְטוֹבָה.
