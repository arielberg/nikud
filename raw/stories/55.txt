ה' צִוָּה אוֹתָנוּ לַעֲשׂוֹת חֶסֶד אֶחָד עִם הַשֵּׁנִי.. אֲבָל, צְרִיכִים לְהִזָּהֵר מְאֹד שֶׁכְּשֶׁמְּבַקְּשִׁים מִמִּישֶׁהוּ אֵיזֶה דָבָר, חַיָּבִים לְהַקְפִּיד לְהַחֲזִיר לוֹ אַחַר כָּךְ.. אִם לְקַחְתֶּם עִפָּרוֹן אוֹ מַחַק חָדָשׁ מֵאֶחָד הַחֲבֵרִים, אַתֶּם צְרִיכִים לוֹמַר לָנוּ וַאֲנַחְנוּ נִקְנֶה עִפָּרוֹן חָדָשׁ כְּדֵי שֶׁתַּחֲזִירוּ לָהֶם.. כִּי אִם לֹא מַחֲזִירִים זֶה נִקְרָא "גֶּזֶל". אִם הֱיִיתֶם בְּאֵיזוֹ חֲנוּת, וּבְטָעוּת הִפַּלְתֶּם אֵיזֶה חֵפֶץ וְהוּא נִשְׁבַּר.. אֲבָל אַף אֶחָד לֹא רָאָה אֶתְכֶם.. מָה תַעֲשׂוּ? בְּמִקְרֶה כָּזֶה, צְרִיכִים מִיָּד לָלֶכֶת לַמּוֹכֵר וְלוֹמַר לוֹ: "סְלִיחָה אֲדוֹנִי, בְּטָעוּת שָׁבַרְתִּי אֶת הַמּוּצָר הַזֶּה.. כַּמָּה אֲנִי צָרִיךְ לְשַׁלֵּם?" אִם הַמּוֹכֵר יִרְצֶה לְוַתֵּר, הוּא כְּבָר יַגִּיד: "לֹא נוֹרָא.. לֹא צָרִיךְ לְשַׁלֵּם.." אֲבָל אִם הוּא יִרְצֶה תַשְׁלוּם, חַיָּבִים לְשַׁלֵּם לוֹ כַּמָּה שֶׁזֶּה עוֹלֶה.. כִּי אַחֶרֶת זֶה "גֶּזֶל", זֶה כְּמוֹ לִגְנֹב.. וְה' צִוָּה אוֹתָנוּ שֶׁאָסוּר לִגְנֹב..

וּמַמָּשׁ לִפְנֵי שֶׁנַּתְחִיל אֶת הַסִּפּוּר נְגַלֶּה לָכֶם שֶׁבְּעֶזְרַת ה' כְּשֶׁתִּהְיוּ גְדוֹלִים, וְיִהְיֶה לָכֶם בַּיִת מִשֶּׁלָּכֶם, לִפְעָמִים תַּזְמִינוּ אִישׁ שֶׁיְּתַקֵּן לָכֶם אֶת הַבֶּרֶז, אוֹ אֶת הַחַשְׁמַל.. תַּקְפִּידוּ תָּמִיד לְשַׁלֵּם לוֹ בְּאוֹתוֹ יוֹם מָה שֶׁמַּגִּיעַ לוֹ, כִּי אִם לֹא מְשַׁלְּמִים זֶה חֲמוּר מְאֹד.. וְעַכְשָׁו, נַתְחִיל אֶת הַסִּפּוּר הַמְיֻחָד.. מוּכָנִים?.. יֹפִי.. שְׁשְׁשְׁ...

בָּעִיר צְפַת, לִפְנֵי מֵאוֹת שָׁנִים, חַי הַצַּדִּיק הַגָּדוֹל רַבִּי יִצְחָק לוּרְיָא אַשְׁכְּנַזִּי. אוֹ בְּקִצּוּר: הָאֲרִ"י הַקָּדוֹשׁ. בַּדּוֹר שֶׁלּוֹ, הָיוּ עוֹד הַרְבֵּה צַדִּיקִים קְדוֹשִׁים מְאֹד שֶׁגָּרוּ בִּצְפַת, הֵם הָיוּ מַמָּשׁ כְּמוֹ מַלְאָכִים.. אֲבָל הָאֲרִ"י הַקָּדוֹשׁ הָיָה צַדִּיק גָּדוֹל יוֹתֵר מִכֻּלָּם. בַּהַתְחָלָה, לֹא הִכִּירוּ אוֹתוֹ כָּל כָּךְ, אֲבָל אַחַר כָּךְ כֻּלָּם רָאוּ שֶׁהוּא הֲכִי גָדוֹל וְצַדִּיק.

אֶחָד הַצַּדִּיקִים שֶׁהָיוּ לְיַד הָאֲרִ"י הָיָה הָרַב 'קוֹל בּוֹכִים', כָּךְ קָרְאוּ לוֹ.. הוּא הָיָה צָם כִּמְעַט כָּל יוֹם, מֵהַבֹּקֶר עַד הַלַּיְלָה.. לָמָּה? כְּדֵי שֶׁה' יְכַפֵּר לוֹ עַל הָעֲווֹנוֹת.. וַאֲפִילוּ שֶׁהוּא הָיָה צַדִּיק, הוּא רָצָה שֶׁמַּמָּשׁ לֹא יִהְיוּ לוֹ עֲווֹנוֹת בִּכְלָל.. אֲפִילוּ הֲכִי קְטַנּוֹת..

יוֹם אֶחָד, בִּקֵּשׁ הָאֲרִ"י מֵהַתַּלְמִידִים שֶׁלּוֹ שֶׁיֵּלְכוּ לָרַב 'קוֹל בּוֹכִים' וִיבַקְּשׁוּ מִמֶּנּוּ שֶׁמָּחָר לֹא יַעֲשֶׂה צוֹם, כִּי הָאֲרִ"י רוֹצֶה לַעֲשׂוֹת סְעוּדַת מִצְוָה יַחַד אִתּוֹ. הֵם אָמְרוּ אֶת זֶה לָרַב 'קוֹל בּוֹכִים' וְהוּא אָמַר: "בְּסֵדֶר.." אַתֶּם יוֹדְעִים אֵיפֹה הֵם נִפְגְּשׁוּ? (יְלָדִים - ) הֵם נִפְגְּשׁוּ לְיַד הַקֶּבֶר שֶׁל רַבִּי יְהוּדָה בַּר אִלְעַאי, זֶה נִמְצָא מַמָּשׁ לְיַד צְפַת.. הֵם יָשְׁבוּ שָׁם, וְדִבְּרוּ דִּבְרֵי תּוֹרָה. לְאַחַר הַסְּעוּדָה הָרַב קוֹל בּוֹכִים אָמַר לָאֲרִ"י: "כְּבוֹד הָרַב, אֲנִי רוֹאֶה שֶׁאַתָּה צַדִּיק קָדוֹשׁ, בְּבַקָּשָׁה תֹאמַר לִי מָה אֲנִי צָרִיךְ לַעֲשׂוֹת כְּדֵי לִהְיוֹת צַדִּיק? אֲנִי מִשְׁתַּדֵּל לָצוּם כָּל יוֹם.. אֲבָל אֲנִי לֹא יוֹדֵעַ אִם זֶה מַסְפִּיק?.." הָאֲרִ"י אָמַר לוֹ: "אִם אֲנִי אֲגַלֶּה לְךָ, אַתָּה מַבְטִיחַ לַעֲשׂוֹת?" וְהָרַב אָמַר: "כֵּן, בְּוַדַּאי! אִם שׁוֹאֲלִים אֶת הָרַב שְׁאֵלָה, צְרִיכִים לַעֲשׂוֹת מָה שֶׁהוּא אוֹמֵר.." וְהָאֲרִ"י אָמַר לוֹ: "טוֹב, אָז תֵּדַע לְךָ שֶׁהַנְּשָׁמָה שֶׁלְּךָ נְקִיָּה, וְאַתָּה בִּכְלָל לֹא צָרִיךְ לָצוּם.. אֶלָּא לְהֵפֶךְ! אַתָּה צָרִיךְ לֶאֱכוֹל כָּל יוֹם אֹכֶל בָּרִיא כְּדֵי שֶׁתִּהְיֶה חָזָק לִלְמוֹד תּוֹרָה.. כִּי ה' כָּל כָּךְ שָׂמֵחַ לִשְׁמוֹעַ מִמְּךָ וּמִכָּל יְהוּדִי דִּבְרֵי תּוֹרָה.." אַחַר כָּךְ, הִגִּיעַ רַב אַחֵר צַדִּיק גָּדוֹל שֶׁקָּרְאוּ לוֹ: רַבִּי מֹשֶׁה גַּלַנְטִי, וְהוּא שָׁאַל אֶת הָאֲרִ"י: "וּמָה אִתִּי? הַאִם יֵשׁ לִי אֵיזֶה עֲווֹן?" הָאֲרִ"י הִסְתַּכֵּל עַל הַמֵּצַח שֶׁלּוֹ אָמַר לוֹ: "כְּבוֹד הָרַב, אַתָּה צַדִּיק גָּדוֹל.. אֲבָל אֲנִי רוֹאֶה שֶׁיֵּשׁ לְךָ עֲווֹן שֶׁל "גֶּזֶל".." רַבִּי מֹשֶׁה הִתְחִיל לִבְכּוֹת? "מָה? יֵשׁ לִי עֲווֹן שֶׁל גָּזָל? אוֹי וַאֲבוֹי.. אֲבָל אֲנִי אַף פַּעַם לֹא לָקַחְתִּי לְמִישֶׁהוּ מַשֶּׁהוּ בְּלִי רְשׁוּת?.. אֵיךְ יָכוֹל לִהְיוֹת?.." הוּא חָזַר לַבַּיִת שֶׁלּוֹ, יָשַׁב עַל הָרִצְפָּה וְהִתְחִיל לִבְכּוֹת..

הַיְּלָדִים שֶׁלּוֹ בָּאוּ אֵלָיו וְשָׁאֲלוּ: "אַבָּא, מָה קָרָה לְךָ? לָמָּה אַתָּה בּוֹכֶה?.." וְהוּא אָמַר לָהֶם: "הַיּוֹם נוֹדַע לִי שֶׁיֵּשׁ לִי עֲווֹן גֶּזֶל.. אֲנִי לֹא יוֹדֵעַ מָה לַעֲשׂוֹת? אֲנִי לֹא רוֹצֶה שֶׁיִּהְיֶה לִי שׁוּם עֲווֹן.. אֲנִי רוֹצֶה לִהְיוֹת הֲכִי צַדִּיק לִכְבוֹד ה'!!" הַיְּלָדִים שֶׁלּוֹ חָשְׁבוּ וְחָשְׁבוּ.. וְאָז אָמְרוּ לוֹ: "אַבָּא, יֵשׁ לָנוּ רַעֲיוֹן.. הֲרֵי יֵשׁ לְךָ מִפְעָל שֶׁל בְּגָדִים, וּבַמִּפְעָל יֵשׁ פּוֹעֲלִים וּפוֹעֲלוֹת שֶׁעוֹבְדִים.. אוּלַי אַתָּה חַיָּב לְאֶחָד מֵהֶם כֶּסֶף וְלֹא שִׁלַּמְתָּ לוֹ.. תִּקְרָא לָהֶם וּתְשַׁלֵּם לָהֶם.."

רַבִּי מֹשֶׁה קָפַץ מִשִּׂמְחָה, הוּא קָרָא לְכָל הָעוֹבְדִים וְאָמַר: "אִם יֵשׁ כָּאן מִישֶׁהוּ שֶׁאֲנִי חַיָּב לוֹ כֶּסֶף שֶׁיַּגִּיד.. אֲנִי רוֹצֶה לְשַׁלֵּם לוֹ.." אֲבָל אַף אֶחָד לֹא דִּבֵּר.. מָה הוּא עָשָׂה? הוּא לָקַח קְעָרָה, שָׂם בָּהּ הַרְבֵּה מַטְבְּעוֹת שֶׁל כֶּסֶף וְאָמַר: "אִם יֵשׁ כָּאן מִישֶׁהוּ שֶׁאֲנִי חַיָּב לוֹ, שֶׁיָּבוֹא לְבַד לַקְּעָרָה עִם הַכֶּסֶף וְיִקַּח כַּמָּה שֶׁאֲנִי חַיָּב לוֹ.." לְפֶתַע, בָּאָה אִשָּׁה אַחַת שֶׁהָיְתָה עוֹבֶדֶת בַּמִּפְעָל, הִתְקָרְבָה לַקְּעָרָה, לָקְחָה שְׁתֵּי מַטְבְּעוֹת קְטַנּוֹת וְהָלְכָה..

רַבִּי מֹשֶׁה, הָלַךְ לָאֲרִ"י הַקָּדוֹשׁ, אֵיךְ שֶׁהוּא נִכְנַס לְפֶתַח הַבַּיִת הָאֲרִ"י אָמַר לוֹ: "כָּל הַכָּבוֹד!! אֲנִי רוֹאֶה שֶׁהֶחֱזַרְתָּ אֶת הַגֶּזֶל!! עַכְשָׁו, אַתָּה צַדִּיק בְּלִי שׁוּם עֲווֹן.. כִּי הָאִשָּׁה הַזּוֹ הָיְתָה מְכִינָה בְּגָדִים יוֹתֵר טוֹב מִכֻּלָּם, וְלָכֵן הָיָה מַגִּיעַ לָהּ מַשְׂכֹּרֶת יוֹתֵר מִכֻּלָּם.. וְעַכְשָׁו שֶׁהִיא לָקְחָה אֶת זֶה - אֵין לְךָ עֲווֹן.."

זֶהוּ צַדִּיקִים, מִכָּאן נִלְמַד לְהִזָּהֵר תָּמִיד מִ"גֶּזֶל" וְלָתֵת לְכָל אֶחָד מָה שֶׁמַּגִּיעַ לוֹ..
