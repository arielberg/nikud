ווֹאוּ כְּבָר שֶׁבַע בַּבֹּקֶר יֵשׁ אוֹר בַּחוּץ אַשְׁכָּרָה עָשִׂינוּ לַיְלָה לָבָן. אַתְּ מְרֻצָּה מֵהַצִּלּוּמִים? בֶּטַח אַתְּ לֹא מַאֲמִינָה לִי לְכָל הַסִּפּוּרִים. אֲנִי יוֹדֵעַ שֶׁאֲנִי חַי בְּסֶרֶט אֲבָל אֵלּוּ הַחַיִּים שֶׁלִּי. אֲנִי מִצְטַעֵר שֶׁדִּבַּרְתִּי בְּרֶצֶף כָּל הַזְּמַן בְּלִי הַפְסָקַת פִּרְסֹמוֹת אֲפִלּוּ אֲבָל אַל תַּחְשְׁבִי שֶׁאֲנִי נַרְקְסִיסְט שֶׁאֲנִי מְרֻכָּז רַק בְּעַצְמִי. בְּהֶחְלֵט יֵשׁ פַּרְשָׁנוּת כָּזוֹ יֵשׁ אֲנָשִׁים שֶׁחוֹשְׁבִים כָּךְ, אֲנִי לֹא שׁוֹלֵל אֶת הַדֵּעָה הַזֹּאת. הִיא בְּהֶחְלֵט תֻּזְכַּר בַּבִּיוֹגְרָפְיָה שֶׁיִּכְתְּבוּ עָלַי. אֲבָל אֲנִי חַיָּב לוֹמַר לָךְ שֶׁיֵּשׁ לָךְ עֵינַיִם יָפוֹת. טוֹב, אַל תִּסְתַּכְּלִי עָלַי כָּכָה. אֲנִי בְּסַךְ הַכֹּל מְנַסֶּה לְסַיֵּעַ לַהִיסְטוֹרְיָה שֶׁתְּתַעֵד אֶת הַבִּיוֹגְרָפְיָה שֶׁלִּי. בְּעִקָּרוֹן אֲנִי יֶלֶד טוֹב. אַתְּ מְבִינָה? כְּשֶׁאֲנִי עָף עַל עַצְמִי אַתְּ צְרִיכָה לְהִתְמוֹגֵג וְלוֹמַר לִי כַּמָּה אֲנִי חָמוּד וּמָתוֹק. בְּבַקָּשָׁה, תּוֹדִי שֶׁאֲנִי יֶלֶד טוֹב כִּכְלוֹת הַכֹּל. אֲנִי קְצָת שׁוֹבָב, אֲבָל אֲנִי רָגִישׁ. אֲנִי חַיָּב לְהַלֵּל אֶת עַצְמִי כְּדֵי לְחַפּוֹת עַל הַחֹסֶר בִּטָּחוֹן שֶׁלִּי. יֵשׁ לָךְ מֻשָּׂג מָה אֲנִי עָבַרְתִּי בַּחַיִּים? אַתְּ צְרִיכָה לְהָבִין אוֹתִי, לְהַבִּיעַ אֶמְפַּתְיָה. אֲבָל אַתְּ יוֹדַעַת מָה? פָאק אִיט. אֲנִי נַרְקְסִיסְט וְנִמְאָס לִי לְהִתְנַצֵּל עַל כָּךְ. אֲנִי מְרֻכָּז בְּעַצְמִי וְרַק בְּעַצְמִי. לֹא מְעַנְיֵן אוֹתִי שׁוּם דָּבָר מִלְּבַד עַצְמִי. הָעוֹלָם לֹא קַיָּם מִבְּחִינָתִי. עַל הַזַּיִן שֶׁלִּי אַתְּ וְהָעוֹלָם וְכֻלָּם. וְלֹא, זֶה לֹא שֶׁלֹּא אִכְפַּת לִי. אֲבָל כְּאָמָּן אֲנִי חַיָּב לְהִתְרַכֵּז בְּעוֹלָמִי הַפְּנִימִי. זֶה הַנּוֹף הַיָּחִיד שֶׁאֲנִי יוֹדֵעַ לְצַיֵּר. מִבְּחִינָתִי אַתְּ לֹא יֵשׁוּת בַּעֲלַת מַשְׁמָעוּת בָּעוֹלָם, אֶלָּא אִם כֵּן אַתְּ שַׁיֶּכֶת לָעוֹלָם שֶׁלִּי בְּאֵיזוֹשֶׁהִי צוּרָה. לְמָשָׁל אִם נַעֲשֶׂה אַהֲבָה. אֲבָל תַּגִּידִי מָה בֶּאֱמֶת אַתְּ חוֹשֶׁבֶת עָלַי? יֵשׁ כְּבָר אוֹר בַּחוּץ ווֹאוּ אַתְּ רוֹצָה לִשְׁתּוֹת קָפֶה? לֹא בְּאֱמֶת אִכְפַּת לִי הָאֱמֶת. אֲבָל חָשׁוּב לִי שֶׁתָּבִינִי שֶׁאֲנִי גָּאוֹן. לֹא סְתָם מְשֻׁגָּע. אֲבָל הַטְּרָגֶדְיָה הִיא שֶׁאֲנִי לֹא יוֹדֵעַ מִי אֲנִי. אַתְּ מְבִינָה? אוּלַי יֵשׁ לִי פִּצּוּל אִישִׁיּוּת. הַפְּסִיכוֹלוֹגִית שֶׁלִּי אָמְרָה שֶׁהַזֶּהוּת שֶׁלִּי מְרֻסֶּקֶת. רֶצֶף הַחַיִּים שֶׁלִּי קָטוּעַ וּמִשְׁתַּנֶּה. כְּאִלּוּ הַבִּיוֹגְרָפְיָה שֶׁלִּי הִיא אוֹסֵף סִפּוּרִים שׁוֹנִים עַל דְּמֻיּוֹת מוּזָרוֹת שֶׁלֹּא קְשׁוּרוֹת אַחַת לַשְּׁנִיָּה, אִלּוּלֵי כֻּלָּן הָיוּ אֲנִי. מְקֻבָּלִים אָמְרוּ שֶׁהִתְגַּלְגְּלוּ בִּי כַּמָּה נְשָׁמוֹת. מִי יוֹדֵעַ. יִתָּכֵן שֶׁאַתְּ צוֹדֶקֶת שֶׁאֲנִי סְתָם מְשֻׁגָּע. אֲנִי לֹא מִתְוַכֵּחַ אִתָּךְ. אֲנִי מַצִּיעַ שֶׁהַהִיסְטוֹרְיָה תִּשְׁפֹּט. מָה אַתְּ חוֹשֶׁבֶת, כְּשֶׁוַאן גוֹך חַי מִישֶׁהוּ יָדַע מִמֶּנּוּ בִּכְלָל? אָמָּנִים גְּדוֹלִים מַקְדִּימִים אֶת הַהִיסְטוֹרְיָה. וְהַמּוֹנָה לִיזַה, אִלּוּ חָיָה בְּיָמֵינוּ אַתְּ חוֹשֶׁבֶת שֶׁהִיא הָיְתָה נֶחְשֶׁבֶת לְיָפָה? מַמָּשׁ לֹא. אַתְּ הַרְבֵּה יוֹתֵר שָׁוָה מִמֶּנָּה. אֲבָל יֹפִי הוּא יַחֲסִי וּמִשְׁתַּנֶּה. הַסִּבָּה הָאֲמִתִּית שֶׁדִּבַּרְתִּי כָּל הַזְּמַן וְלֹא הִקְשַׁבְתִּי לָךְ כִּי בְּכָל מִקְרֶה לֹא הָיִיתִי מְסֻגָּל לְהַקְשִׁיב לָךְ. קָשֶׁה לִי לְהִתְרַכֵּז מוּל בַּחוּרָה יָפָה. אַתְּ מְבִינָה? תַּחְשְׁבִי לְמָשָׁל עַל סֶרְג' גִינְסְבּוּרְג וְבְּרִיזִ'יט בַּארְדוֹ. מָה אַתְּ לֹא יוֹדַעַת מִי זֶה? תִּתְבַּיְּשִׁי לָךְ. הוּא הָיָה יְהוּדִי מְכֹעָר, אֲבָל הוּא יָצָא עִם הַנָּשִׁים הֲכִי יָפוֹת בְּצָרְפַת. וְזֶה מַמָּשׁ מַחְשִׁיד בְּעֵינַי שֶׁאַתְּ נוֹסַעַת עִם הַהוֹרִים שֶׁלָּךְ לְסִין. זֶה לֹא מוֹצֵא חֵן בְּעֵינַי. מָה לָמָּה? זֶה מַמָּשׁ קְרִיפִּי. בַּת כַּמָּה אַתְּ? תַּתְחִילִי לְהִתְבַּגֵּר. בַּגִּיל שֶׁלָּךְ אַתְּ חַיֶּבֶת לִמְרֹד, לִבְעֹט, לְחַפֵּשׂ אֶת עַצְמֵךְ, אַתְּ לֹא יְכוֹלָה סְתָם לִנְסֹעַ עִם הוֹרִים שֶׁלָּךְ לְסִין. אַתְּ חוֹשֶׁבֶת שֶׁאֲנִי מְקַנֵּא. אֲבָל מַמָּשׁ לֹא. זֶה מַמָּשׁ יַלְדוּתִי בְּעֵינַי. אֲבָל מָה אִכְפַּת לִי, תִּסְּעִי עִם הוֹרִים שֶׁלָּךְ לְסִין. מִצִּדִּי תִּסְּעִי מַמָּשׁ עַכְשָׁו. וְאַל תַּחְזְרִי אַף פַּעַם. אַתְּ פָּשׁוּט יַלְדַּת שַׁמֶּנֶת מְפֻנֶּקֶת. כָּכָה לְעוֹלָם לֹא תִּהְיִי אָמָּנִית. אַתְּ רוֹצָה שֶׁאֲנִי אַגִּיד לָךְ אֶת הָאֱמֶת? אֲנִי יָכוֹל לוֹמַר לָךְ עַכְשָׁו שֶׁכָּל מָה שֶׁסִּפַּרְתִּי לָךְ סְתָם מֻמְצָא. אֲבָל אֲנִי לֹא שַׁקְרָן. הַתַּפְקִיד שֶׁלִּי כְּאָמָּן הוּא לְסַפֵּר סִפּוּרִים. מָה שַׁ"י עַגְנוֹן לֹא הִמְצִיא שֶׁהוּא נוֹלַד בְּט' בָּאָב? כְּשֶׁאַתְּ רוֹאָה סֶרֶט מְעַנְיֵן אוֹתָךְ הַאִם הַסִּפּוּר אֲמִתִּי? לֹא. בְּאִמָּשְׁלָּךְ. אֲבָל לַמְרוֹת זֹאת הַכֹּל אֱמֶת. יֵשׁ לִי תֵּעוּד שֶׁל הַכֹּל, בְּחַיָּי. יוֹאוּ, אַתְּ יוֹדַעַת פַּעַם עָצְרָה אוֹתִי נַיֶּדֶת בְּאֶמְצַע הַלַּיְלָה בַּצָּפוֹן וְהָיָה לִי עָלַי קוֹק. כְּשֶׁחָקְרוּ אוֹתִי בַּתַּחֲנָה סִפַּרְתִּי לָהֶם אֶת סִפּוּר חַיָּי, כָּל הַשּׁוֹטְרִים יָשְׁבוּ מְרֻתָּקִים. מְפַקֵּד הַתַּחֲנָה הֵכִין לִי סֶנְדְוִיץ'. לֹא פָּתְחוּ לִי תִּיק וְלֹא נַעֲלַיִם. הוּא אָמַר לִי שֶׁאֲנִי יֶלֶד טוֹב, שֶׁאֲנִי אֶשְׁמֹר עַל עַצְמִי לֹא לְהִסְתַּבֵּךְ. אַתְּ רוֹאָה? אֲפִלּוּ הַשּׁוֹטֵר אָמַר שֶׁאֲנִי יֶלֶד טוֹב. וְגַם כָּתַבְתִּי עַל זֶה שִׁיר. זֶה כָּתוּב בַּסֵפֶר! תִּפְתְּחִי אֶת הַסֵּפֶר, נוּ בְּבַקָּשָׁה. אֲנִי רוֹצָה שֶׁתַּקְרִיאִי אֶת הַשִּׁיר. אַתְּ חַיֶּבֶת לְהַאֲמִין לִי שֶׁאֲנִי יֶלֶד טוֹב. וְגַם אִם לֹא תַּאֲמִינִי לִי, לֹא אִכְפַּת לִי בִּכְלָל. אַתְּ הַבַּחוּרָה הֲכִי מוּזָרָה שֶׁפָּגַשְׁתִּי אֵי פַּעַם. אֲנִי לֹא מֵבִין לָמָּה אֲנִי בִּכְלָל נִפְתַּח בְּפָנַיִךְ. יוֹאוּ בֶּטַח אַתְּ חוֹשֶׁבֶת שֶׁאַתְּ מַכִּירָה אוֹתִי אֲבָל אַל תַּשְׁלִי אֶת עַצְמֵךְ. אֵין לָךְ מֻשָּׂג בִּכְלָל מִי אֲנִי. לְאַף אֶחָד אֵין מֻשָּׂג. לֹא מְשַׁנֶּה, תַּחְשְׁבִי מָה הָיָה יָכוֹל לִהְיוֹת אִלּוּ אֲנִי וְיוֹנָה ווֹלָךְ הַיְנוּ חַיִּים בְּאוֹתָהּ תְּקוּפָה. אֲנַחְנוּ הֲרֵי נְשָׁמוֹת תְּאוֹמוֹת. הַרְבֵּה מֵהַשִּׁירִים שֶׁלָּהּ הִיא כָּתְבָה עָלַי. ווּאַי כְּבָר מְאֻחָר. אַתְּ יְכוֹלָה לָלֶכֶת לֹא אִכְפַּת לִי. הָיִיתִי אוֹמֵר לָךְ תּוֹדָה עַל הַהַקְשָׁבָה שֶׁבַע שָׁעוֹת, אֲבָל בֵּינֵינוּ הָעֹנֶג הָיָה כֻּלּוֹ שֶׁלָּךְ. טוֹב בְּבַקָּשָׁה אַל תֵּלְכִי. אֲנִי אֶהְיֶה מְאוֹד עָצוּב בְּבַקָּשָׁה אַל תַּשְׁאִירִי אוֹתִי לְבַד. הַחָבֵר הֲכִי טוֹב שֶׁלִּי מֵת שָׁנָה שֶׁעָבְרָה מִקּוֹר. פֹּה מַמָּשׁ בַּשּׁוּק. אַל תֵּלְכִי אֲנִי אֶהְיֶה יֶלֶד טוֹב.
