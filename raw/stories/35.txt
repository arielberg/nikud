אַתֶּם צְרִיכִים לָדַעַת שֶׁה' נָתַן לְעַם יִשְׂרָאֵל שְׁלוֹשָׁה כְּתָרִים. מַה הֵם שְׁלוֹשֶׁת הַכְּתָרִים כֶּתֶר אֶחָד הוּא כֶּתֶר מַלְכוּת, הַכֶּתֶר הַשֵּׁנִי הוּא כֶּתֶר כְּהוּנָה, וְהַכֶּתֶר הַשְּׁלִישִׁי הוּא כֶּתֶר תּוֹרָה. אֶת הַכֶּתֶר הָרִאשׁוֹן שֶׁל הַמַּלְכוּת ה' נָתַן רַק לְדָוִד הַמֶּלֶךְ וְלַבָּנִים שֶׁלּוֹ, רַק הֵם יְכוֹלִים לִהְיוֹת הַמְּלָכִים שֶׁל עַם יִשְׂרָאֵל וְגַם מֶלֶךְ הַמָּשִׁיחַ יִהְיֶה מֵהַבָּנִים שֶׁל דָּוִד הַמֶּלֶךְ. אֶת הַכֶּתֶר הַשֵּׁנִי שֶׁהוּא כֶּתֶר הַכְּהוּנָה ה' נָתַן לְאַהֲרֹן הַכֹּהֵן וְלַבָּנִים שֶׁלּוֹ, רַק הֵם יְכוֹלִים לִהְיוֹת כֹּהֲנִים וְלַעֲבֹד בְּבֵית הַמִּקְדָּשׁ. אֲבָל יְלָדִים יְקָרִים, יֵשׁ לָנוּ הַפְתָּעָה בִּשְׁבִילְכֶם, מָה הַהַפְתָּעָה? הַקְשִׁיבוּ טוֹב, אֶת הַכֶּתֶר הַשְּׁלִישִׁי שֶׁהוּא כֶּתֶר הַתּוֹרָה, ה' לֹא נָתַן לְמִישֶׁהוּ מְסֻיָּם, אֶלָּא כָּל אֶחָד מֵעַם יִשְׂרָאֵל, גַּם אֲנַחְנוּ וְגַם אַתֶּם, גַּם הַשְּׁכֵנִים וְהַחֲבֵרִים שֶׁלָּנוּ וְשֶׁלָּכֶם, וְכָל מִי שֶׁיִּרְצֶה, יָכוֹל לָבוֹא וְלָקַחַת אֶת הַכֶּתֶר הַזֶּה. אַתֶּם יוֹדְעִים אֵיךְ? פָּשׁוּט, עַל יְדֵי שֶׁלּוֹמְדִים אֶת הַתּוֹרָה. וְעַכְשָׁו נִשְׁאַל אֶתְכֶם חִידָה: אֵיזֶה כֶּתֶר מִבֵּין שְׁלֹשְׁתָּם הוּא הַכֶּתֶר הֲכִי גָדוֹל?  נְגַלֶּה לָכֶם שֶׁהַכֶּתֶר הֲכִי גָדוֹל וְחָשׁוּב הוּא: כֶּתֶר הַתּוֹרָה!! וְעַכְשָׁו, נַתְחִיל אֶת הַסִּפּוּר הַמְיֻחָד..

לִפְנֵי שָׁנִים רַבּוֹת, חַי בְּבָבֶל צַדִּיק שֶׁקָּרְאוּ לוֹ: מַר חֲנִינָא. מַר חֲנִינָא הָיָה אַחֲרַאי עַל הַקְּהִלָּה שֶׁל הַיְּהוּדִים בְּבָבֶל, הוּא גַם הָיָה תַלְמִיד חָכָם וְגַם הָיָה מִזַּרְעוֹ שֶׁל דָּוִד הַמֶּלֶךְ. לְמַר חֲנִינָא הָיָה בֶּן שֶׁקָּרְאוּ לוֹ: מַר הוּנָא וְהוּא הָיָה עוֹזֵר לְאַבָּא שֶׁלּוֹ בְּכָל דָּבָר. בְּאוֹתוֹ זְמַן, הָיָה בְּבָבֶל מֶלֶךְ גּוֹי שֶׁקָּרְאוּ לוֹ: "כּוּזְרוֹי". יוֹם אֶחָד, שָׁלַח כּוּזְרוֹי מִכְתָּב לְמַר חֲנִינָא וְקָרָא לוֹ לָאַרְמוֹן. אֲפִילוּ שְׁמָר חֲנִינָא הָיָה כְּבָר מְבֻגָּר, בְּכָל זֹאת הוּא כִּבֵּד הַמֶּלֶךְ וּבָא אֵלָיו, כְּשֶׁהִגִּיעַ לָאַרְמוֹן שָׁאַל אֶת הַמֶּלֶךְ "מַדּוּעַ כְּבוֹד הַמֶּלֶךְ הִזְמִין אוֹתִי אֵלָיו?" וְהַמֶּלֶךְ כּוּזְרוֹי אָמַר לוֹ: "תִּשְׁמַע מַר חֲנִינָא, אֲנִי עוֹמֵד מוּל מִלְחָמָה קָשָׁה מְאֹד. אֲנִי הוֹלֵךְ לְהִלָּחֵם עִם מֶלֶךְ רוֹמָא שֶׁקּוֹרְאִים לוֹ "כּוּזְלָא"" מַר חֲנִינָא אָמַר: "אֲנִי מְאַחֵל לַמֶּלֶךְ הַצְלָחָה רַבָּה, אֲבָל מַדּוּעַ כְּבוֹד הַמֶּלֶךְ הִזְמִין אוֹתִי אֵלָיו?" וְהַמֶּלֶךְ אָמַר: "אֲנִי רוֹצֶה שֶׁגַּם הַיְּהוּדִים יָבוֹאוּ וְיַעַזְרוּ לִי בַּמִּלְחָמָה! אִם תַּעַזְרוּ לִי לְנַצֵּחַ אוֹתוֹ, אֶתֵּן לָכֶם בְּמַתָּנָה אֶת אֶרֶץ יִשְׂרָאֵל!!" מַר חֲנִינָא מְאֹד הִתְרַגֵּשׁ.. הוּא כָּל כָּךְ רָצָה אֶת אֶרֶץ יִשְׂרָאֵל.. הוּא הָלַךְ הַבַּיְתָהּ וְדִבֵּר עִם הַבֶּן שֶׁלּוֹ: מַר הוּנָא. מַר הוּנָא שָׂמַח וְהִצְלִיחַ לְגַיֵּס שְׁלוֹשֶׁת אֲלָפִים בַּחוּרִים כְּדֵי לְהִלָּחֵם בְּמֶלֶךְ רוֹמָא. הֵם יָצְאוּ וְנִלְחֲמוּ בְּכָל הַכֹּחַ, וְכַמּוּבָן הִתְפַּלְּלוּ לַה' שֶׁיַּעֲזֹר לָהֶם, וּבִזְכוּת הַתְּפִלּוֹת וּמְסִירוּת הַנֶּפֶשׁ שֶׁלָּהֶם הֵם נִצְּחוּ!!

הֵם חָזְרוּ בְּשִׂמְחָה לְמֶלֶךְ בָּבֶל וְאָמְרוּ לוֹ: "בָּרוּךְ ה', נִצַּחְנוּ אֶת הָאוֹיְבִים, וְעַכְשָׁו תְּקַיֵּם אֶת הַהַבְטָחָה שֶׁלְּךָ וְתִתֵּן לָנוּ לִשְׁלוֹט בְּאֶרֶץ יִשְׂרָאֵל". אֲבָל הַמֶּלֶךְ כּוּזְרוֹי לֹא קִיֵּם אֶת הַהַבְטָחָה שֶׁלּוֹ, הוּא אָמַר לָהֶם: "נָכוֹן שֶׁאַתֶּם נִצַּחְתֶּם, אֲבָל תַּגִּידוּ תוֹדָה שֶׁאַתֶּם גָּרִים אֶצְלִי בְּבָבֶל, אֲנִי לֹא צָרִיךְ לָתֵת לָכֶם גַּם אֶת אֶרֶץ יִשְׂרָאֵל.." הַבַּחוּרִים הַיְּהוּדִים שֶׁנִּלְחֲמוּ, כָּעֲסוּ נוֹרָא... אַתֶּם יוֹדְעִים מָה עָשׂוּ? הֵם הִתְחִילוּ לְהִלָּחֵם אִתּוֹ.. אוֹי וַאֲבוֹי! כּוּזְרוֹי כָּעַס נוֹרָא.. כְּשֶׁהוּא רָאָה שֶׁהֵם נִלְחָמִים בּוֹ הוּא קָרָא לְאַלְפֵי חַיָּלִים שֶׁלּוֹ וְנִלְחַם בָּהֶם וְכַמּוּבָן שֶׁהוּא נִצֵּחַ אוֹתָם.. אֲבָל הוּא לֹא הִסְתַּפֵּק בְּזֶה.. אֶלָּא הוּא צִוָּה לַהֲרוֹג אֶת מַר הוּנָא וְאֶת כָּל הַמִּשְׁפָּחָה שֶׁלּוֹ. וְזֶה הָיָה אָסוֹן גָּדוֹל, לָמָּה? כִּי רַק הַמִּשְׁפָּחָה שֶׁל מַר הוּנָא נִשְׁאֲרָה מִדָּוִד הַמֶּלֶךְ, וְאִם חַס וְשָׁלוֹם כֻּלָּם יָמוּתוּ אֵיךְ יִוָּלֵד מֶלֶךְ הַמָּשִׁיחַ?? אֲבָל אַל תִּדְאֲגוּ, ה' תָּמִיד שׁוֹמֵר עַל הַמִּשְׁפָּחָה שֶׁל דָּוִד הַמֶּלֶךְ.. אַתֶּם יוֹדְעִים מָה קָרָה? מִכָּל הַמִּשְׁפָּחָה שֶׁל מַר הוּנָא הִצְלִיחָה אִשָּׁה אַחַת לִבְרֹחַ, מִי זוֹ? "יַלְתָּא" אִשְׁתּוֹ שֶׁל מַר הוּנָא, הִיא בָּרְחָה מֵהַחַיָּלִים וְהִתְחַבְּאָה אֵצֶל אַבָּא שֶׁלָּה שֶׁקָּרְאוּ לוֹ "רַבִּי חוֹפְנִי", וְשָׁם נוֹלַד לָהּ בֶּן קָטָן.

כּוּזְרוֹי לֹא יָדַע שֶׁהִיא בָּרְחָה וְהוּא יָשַׁב בָּאַרְמוֹן שֶׁלּוֹ שָׂמֵחַ. אֲבָל, פִּתְאוֹם קָרָה לוֹ מַשֶּׁהוּ מְאֹד מוּזָר. מָה קָרָה? כּוּזְרוֹי הִתְעוֹרֵר בַּבֹּקֶר וְרָאָה שֶׁיֵּשׁ לוֹ דָם עַל הַמֵּצַח. "אַי, מָה זֶה? מָה קָרָה לִי? מֵאֵיפֹה יֵשׁ לִי דָם בַּמֵּצַח?.." כּוּזְרוֹי נִבְהַל.. וְאָז הוּא נִזְכַּר שֶׁבַּלַּיְלָה הוּא חָלַם אֵיזֶה חֲלוֹם אֲבָל הוּא לֹא הִצְלִיחַ לְהִזָּכֵר מָה הָיָה בְּדִיּוּק הַחֲלוֹם.. הוּא שָׁאַל אֶת הָעֲבָדִים שֶׁלּוֹ אֲבָל אַף אֶחָד לֹא יָכַל לָדַעַת.. כָּל הַמְּדִינָה דִבְּרָה עַל הַחֲלוֹם שֶׁל הַמֶּלֶךְ עַד שֶׁרַבִּי חוֹפְנִי שָׁמַע אֶת זֶה וְאָמַר לָעֲבָדִים שֶׁלּוֹ: "תַּגִּידוּ לַמֶּלֶךְ שֶׁאֲנִי יָכוֹל לְהַזְכִּיר לוֹ מָה הָיָה הַחֲלוֹם.."

רַבִּי חוֹפְנִי הִגִּיעַ לַמֶּלֶךְ וְאָמַר לוֹ: "אֲדוֹנִי הַמֶּלֶךְ, אֲנִי אַזְכִּיר לְךָ מָה חָלַמְתָּ. אַתָּה חָלַמְתָּ שֶׁהָיִיתָ מְטַיֵּל עַל שְׂפַת הַנָּהָר, זֶה הָיָה בִּשְׁעוֹת הַבֹּקֶר וְהָיָה נוֹף יָפֶה מְאֹד.." הַמֶּלֶךְ צָעַק: "כֵּן, כֵּן.. אֲנִי מַתְחִיל לְהִזָּכֵר.. זֶה הָיָה מָקוֹם מְאֹד יָפֶה.. תַּמְשִׁיךְ לְהַזְכִּיר לִי.. בְּבַקָּשָׁה.." וְרַבִּי חוֹפְנִי הִמְשִׁיךְ: "הָיָה שָׁם גַּן יָפֶה מְאֹד שֶׁנִּקְרָא "בֻּסְתָּן", וְהָיוּ בּוֹ עֵצִים וּפְרָחִים מְאֹד יָפִים.. שָׁאַלְתָּ אֶת הַשּׁוֹמֵר שֶׁהָיָה שָׁם: "שֶׁל מִי הַגַּן הַזֶּה?" וְהַשּׁוֹמֵר אָמַר לְךָ שֶׁהוּא שַׁיָּךְ לְאָדָם אֶחָד שֶׁאַתָּה לֹא מַכִּיר.. וְאָז, כָּעַסְתָּ מְאֹד שֶׁהַגַּן לֹא שֶׁלְּךָ, לָקַחְתָּ גַּרְזֶן וְהִתְחַלְתָּ לַהֲרֹס אֶת כָּל הָעֵצִים וְהַפְּרָחִים.. עַד שֶׁנִּשְׁאַר רַק פֶּרַח אֶחָד אֲבָל אַתָּה רָצִיתָ לַעֲקוֹר גַּם אוֹתוֹ.. וְאָז, לְפֶתַע, הִגִּיעַ אִישׁ זָקֵן שֶׁנָּתַן לְךָ מַכָּה וְצָעַק עָלֶיךָ: "תַּעֲצֹר! אִם תַהֲרֹס גַּם אֶת הַצֶּמַח הָאַחֲרוֹן, אֲנִי אֶהֱרוֹג אוֹתְךָ!!" הַמֶּלֶךְ פָּרַץ בִּבְכִי וְאָמַר: "נָכוֹן, נָכוֹן.. הִבְטַחְתִּי לוֹ שֶׁלֹּא אֶעֱקֹר אֶת הַצֶּמַח הָאַחֲרוֹן.. וְהִתְעוֹרַרְתִּי.. וְעַכְשָׁו, רַבִּי חוֹפְנִי, אוּלַי תּוּכַל לוֹמַר לִי מָה הַפֵּרוּשׁ שֶׁל הַחֲלוֹם??" וְרַבִּי חוֹפְנִי הִסְבִּיר: "הַגַּן הַזֶּה, זֶה הַמִּשְׁפָּחָה שֶׁל דָּוִד הַמֶּלֶךְ, שֶׁהָרַגְתָּ אֶת כֻּלָּה, וְהַזָּקֵן שֶׁפָּגַשְׁתָּ הָיָה דָוִד הַמֶּלֶךְ. וְהוּא הִזְהִיר אוֹתְךָ שֶׁאִם תַּהֲרֹג גַּם אֶת הַתִּינוֹק הָאַחֲרוֹן שֶׁנִּשְׁאַר - הוּא יַעֲנִישׁ אוֹתְךָ". כְּשֶׁשָּׁמַע הַמֶּלֶךְ אֶת זֶה, הוּא קָרָא לְיַלְתָּא וְלַתִּינוֹק שֶׁלָּה וְגִדֵּל אוֹתוֹ בַּבַּיִת שֶׁלּוֹ, וְאֵיךְ קָרְאוּ לַתִּינוֹק? (יְלָדִים - ) בּוּסְתְּנַאי. וּמֵהַתִּינוֹק הַזֶּה יֵצֵא בְּעֶזְרַת ה' מֶלֶךְ הַמָּשִׁיחַ.
