אַתֶּם צְרִיכִים לָדַעַת, שֶׁזֶּה מְאֹד חָשׁוּב לָדַעַת לְהִתְאַפֵּק. יֵשׁ יְלָדִים שֶׁלֹּא יְכוֹלִים לְהִתְאַפֵּק וְתָמִיד אוֹמְרִים: "לֹא.. עַכְשָׁו! עַכְשָׁו!!" כָּל דָּבָר הֵם רוֹצִים עַכְשָׁו.. וְזֶה לֹא טוֹב. לָמָּה? כִּי כָּל אָדָם חַיָּב שֶׁיִּהְיֶה לוֹ סַבְלָנוּת. לְדֻגְמָא: אִם הוֹלְכִים עִם אַבָּא אוֹ אִמָּא לַמַּכֹּלֶת וְרוֹאִים מַמְתָּק, מְבַקְּשִׁים יָפֶה: " אַבָּא בְּבַקָּשָׁה, תִּקְנֶה לִי אֶת הַמַּמְתָּק.." לִפְעָמִים אַבָּא יָכוֹל לִקְנוֹת וְלִפְעָמִים לֹא.. יֶלֶד שֶׁאֵין לוֹ סַבְלָנוּת מַתְחִיל לִבְכּוֹת וּלְהִשְׁתּוֹלֵל וּלְבַיֵּשׁ אֶת אִמָּא אוֹ אַבָּא לְיַד כֻּלָּם, וְגַם אֶת עַצְמוֹ.. כִּי כֻּלָּם מִסְתַּכְּלִים עָלָיו וְאוֹמְרִים: "אֵיזֶה יֶלֶד זֶה.. אֵין לוֹ בִּכְלָל סַבְלָנוּת.." אֲבָל יֶלֶד שֶׁיֵּשׁ לוֹ סַבְלָנוּת אוֹמֵר: "טוֹב, אֲנִי אֲחַכֶּה לַפַּעַם הַבָּאָה.." וְאָז אַבָּא וְאִמָּא שְׂמֵחִים, וְגַם אֲנָשִׁים מִסְתַּכְּלִים וְאוֹמְרִים: "וּוּאַי, אֵיזֶה יֶלֶד בּוֹגֵר.. הוּא יוֹדֵעַ לְהַמְתִּין.." ה' גַּם רוֹצֶה שֶׁתִּהְיֶה לָנוּ סַבְלָנוּת.. כְּמוֹ לִפְעָמִים שֶׁאוֹכְלִים מַשֶּׁהוּ בְּשָׂרִי, צָרִיךְ לְחַכּוֹת בְּסַבְלָנוּת שֵׁשׁ שָׁעוֹת לִפְנֵי שֶׁאוֹכְלִים מַמְתָּק חֲלָבִי.. וְעַכְשָׁו, נַמְשִׁיךְ אֶת הַסִּפּוּר שֶׁל שְׁלֹמֹה אֶפְרַיִם.. שְׁשְׁשְׁ...

שְׁלֹמֹה אֶפְרַיִם הָיָה יֶלֶד מְיֻחָד וְנִפְלָא, הוּא לֹא יָדַע שֶׁהוּא יְהוּדִי וְהַהוֹרִים שֶׁלּוֹ הֵם רַבִּי אַהֲרֹן וְאִשְׁתּוֹ, אֶלָּא מֵאַחַר וְהוּא גָדַל אֵצֶל הַדֻּכָּס, חָשַׁב שֶׁהוּא גוֹי וְהוּא הַבֶּן שֶׁלּוֹ.. כָּךְ הוּא גָדַל עַד שֶׁהִגִּיעַ כִּמְעַט לְגִיל שְׁלֹשׁ עֶשְׂרֵה, וְאָז קָרָה מַשֶּׁהוּ מוּזָר.. מָה קָרָה? בְּאֶמְצַע הַלַּיְלָה הוּא חָלַם חֲלוֹם. בַּחֲלוֹם, בָּא אֵלָיו אַבָּא שֶׁלּוֹ: רַבִּי אַהֲרֹן וְאָמַר לוֹ: "דַּע לְךָ, שֶׁאַתָּה בִּכְלָל לֹא הַבֶּן שֶׁל הַדֻּכָּס, אַתָּה הַבֶּן שֶׁלִּי!! אַתָּה יְהוּדִי! וְאָסוּר לְךָ לְהִשָּׁאֵר אֵצֶל הַדֻּכָּס יוֹתֵר.." שְׁלֹמֹה אֶפְרַיִם הִתְעוֹרֵר וְנִבְהַל מְאֹד, אֲבָל הוּא חָשַׁב שֶׁזֶּה סְתָם חֲלוֹם..

לְמָחֳרָת בַּלַּיְלָה שׁוּב בָּא אֵלָיו רַבִּי אַהֲרֹן בַּחֲלוֹם וְאָמַר לוֹ: "בְּנִי הַיָּקָר, אַתָּה מִתְקָרֵב לְגִיל שְׁלֹשׁ עֶשְׂרֵה, בַּגִּיל הַזֶּה אַתָּה נִהְיָה בַּר מִצְוָה, וְאַתָּה צָרִיךְ לְהַנִּיחַ תְּפִלִּין, קוּם מַהֵר וְתִבְרַח מֵהַבַּיִת לָעִיר 'פְּרָאג', שָׁם יֵשׁ הַרְבֵּה יְהוּדִים וְהֵם יִדְאֲגוּ לְךָ.." אֲבָל גַּם לַחֲלוֹם הַזֶּה הוּא לֹא הִתְיַחֵס.. רַבִּי אַהֲרֹן הִמְשִׁיךְ לָבוֹא אֵלָיו כָּל לַיְלָה.. וּשְׁלֹמֹה אֶפְרַיִם הִתְחִיל לִהְיוֹת עָצוּב.. הַדֻּכָּס שָׁאַל אוֹתוֹ: "מָה קָרָה לְךָ? אַתָּה לֹא מַרְגִּישׁ טוֹב?.." וְהוּא סִפֵּר לוֹ אֶת הַחֲלוֹם.. הַדֻּכָּס נִבְהַל וְאָמַר: "אַל תָּשִׂים לֵב, זֶה לֹא נָכוֹן.. הֲרֵי אֲנִי אַבָּא שֶׁלְּךָ.." אֲבָל שְׁלֹמֹה אֶפְרַיִם לֹא נִרְגַּע..

כְּשֶׁהִגִּיעַ הַלַּיְלָה הָאַחֲרוֹן לִפְנֵי בַּר הַמִּצְוָה שֶׁלּוֹ, שׁוּב בָּא אֵלָיו רַבִּי אַהֲרֹן בַּחֲלוֹם וְאָמַר לוֹ: "בִּגְלַל שֶׁאַתָּה לֹא מֵבִין.. אֲנִי צָרִיךְ לָקַחַת אוֹתְךָ בְּכֹחַ, אֲנִי אָשִׂים אוֹתְךָ בָּעִיר פְּרָאג לְיַד בֵּית הַכְּנֶסֶת שֶׁל הַמַּהַרַ"ל מִפְּרָאג שֶׁהוּא צַדִּיק גָּדוֹל.. אַתָּה תְסַפֵּר לוֹ מָה שֶׁקָּרָה לְךָ וְהוּא כְּבָר יֵדַע אֵיךְ לִדְאֹג לְךָ.." וְאָז, פִּתְאוֹם שְׁלֹמֹה אֶפְרַיִם הִרְגִּישׁ שֶׁהוּא כְּאִלּוּ עָף בָּאֲוִיר.. וְ.. "בּוּם!" הוּא צָנַח לְיַד בֵּית הַכְּנֶסֶת שֶׁל הַמַּהַרַ"ל מִפְּרָאג.. זֶה הָיָה מֻקְדָּם בַּבֹּקֶר, הָיָה קַר מְאֹד.. הוּא עָמַד לְיַד הַשַּׁעַר וְהַגַּבַּאי שֶׁל בֵּית הַכְּנֶסֶת הִגִּיעַ.. "הֵי, יֶלֶד קָטָן.. מָה אַתָּה עוֹשֶׂה כָּאן?.." שְׁלֹמֹה אֶפְרַיִם הָיָה מְבֻלְבָּל, הוּא אָמַר: "אֲנִי רוֹצֶה לְדַבֵּר עִם הָרַב שֶׁלָּכֶם.." הַגַּבַּאי הֵבִיא אוֹתוֹ לַמַּהַרַ"ל וְהוּא סִפֵּר לוֹ אֶת כָּל הַסִּפּוּר.. הָרַב הֵבִין מִיָּד מָה שֶׁקָּרָה, וְהֵבִיא לוֹ תְפִילִין.. הוּא הִנִּיחַ אֶת הַתְּפִלִּין.. וְהַמַּהַרַ"ל יָשַׁב לִלְמוֹד אִתּוֹ תוֹרָה.. וּבִגְלַל שֶׁשְּׁלֹמֹה אֶפְרַיִם הָיָה יֶלֶד חָכָם מְאֹד.. הוּא הֵבִין אֶת הַתּוֹרָה מְצֻיָּן, וּמִיּוֹם לְיוֹם גָּדַל וְנִהְיָה תַלְמִיד חָכָם...

בֵּינְתַיִם, הַדֻּכָּס קָם בַּבֹּקֶר, הָלַךְ לַחֶדֶר שֶׁל הַיֶּלֶד, וּמָה הוּא רוֹאֶה? נָכוֹן, הַמִּטָּה רֵיקָה!! "אוֹי וַאֲבוֹי!! אֵיפֹה הַיֶּלֶד הַמְּיֻחָד שֶׁלִּי??" הַדֻּכָּס הִתְחִיל לְחַפֵּשׂ אוֹתוֹ בְּכָל הַמְּדִינוֹת.. הוּא הֵבִיא צִיּוּר שֶׁלּוֹ וְשָׁאַל אֶת כֻּלָּם: "הַאִם רְאִיתֶם יֶלֶד כָּזֶה??" עַד שֶׁהוּא הִצְלִיחַ לְהַגִּיעַ לְבֵית הַמִּדְרָשׁ שֶׁבּוֹ הוּא לָמַד.. הוּא נִכְנַס לְשָׁם וְ-חָ-טַ-ף אֶת שְׁלֹמֹה אֶפְרַיִם וְהִכְנִיס אוֹתוֹ לָעֲגָלָה וְהִתְחִיל לִבְרֹחַ מֵהָעִיר.. אֲבָל ה' עָזַר לַמַּהַרַ"ל מִפְּרָאג, וְהוּא הִצְלִיחַ לַעֲצֹר אוֹתָם בַּשַּׁעַר שֶׁל הָעִיר וְלָקַחַת בַּחֲזָרָה אֶת שְׁלֹמֹה אֶפְרַיִם מֵהַדֻּכָּס.. הַמַּהַרַ"ל מִפְּרָאג, פָּחַד מְאֹד עַל שְׁלֹמֹה אֶפְרַיִם.. לָכֵן, מָה הוּא עָשָׂה? הוּא שָׁלַח אוֹתוֹ לִכְפָר אֶחָד רָחוֹק, וְאָמַר לוֹ: "שְׁלֹמֹה אֶפְרַיִם הַיָּקָר, אַתָּה צָרִיךְ לָדַעַת, שֶׁיֵּצֶר הָרָע רוֹצֶה לְהַכְשִׁיל אוֹתְךָ וְלִגְרֹם לְךָ לַחֲזֹר לַדֻּכָּס, וְהַדֻּכָּס רוֹצֶה שֶׁתִּהְיֶה חַס וְשָׁלוֹם גּוֹי.. לָכֵן, אַתָּה צָרִיךְ לָלֶכֶת לִכְפָר רָחוֹק, לֹא לְסַפֵּר לְאַף אֶחָד אֵיךְ קוֹרְאִים לְךָ.. וּלְהִתְנַהֵג כְּמוֹ אָדָם פָּשׁוּט.." שְׁלֹמֹה אֶפְרַיִם הָלַךְ לַכְּפָר שֶׁאָמַר לוֹ הַמַּהַרַ"ל וְשָׁם הִתְחִיל לִמְכֹּר בֵּיצִים. כֻּלָּם קָרְאוּ לוֹ: "אֶפְרַיִם מוֹכֵר בֵּיצִים". אֲבָל בְּכָל פַּעַם כְּשֶׁהָיָה מוֹכֵר לְמִישֶׁהוּ בֵּיצִים הָיָה חוֹתֵם עַל הַדַּף: "אֶפְרַיִם עוֹלְלוֹת". עָבְרוּ כַּמָּה שָׁנִים, וְהַמַּהַרַ"ל מִפְּרָאג הָיָה חוֹלֶה וְעָמַד לָמוּת.. הַתַּלְמִידִים שָׁאֲלוּ אוֹתוֹ: "כְּבוֹד הָרַב, מִי יִהְיֶה הָרַב שֶׁלָּנוּ אַחֲרֵי שֶׁאַתָּה תִפָּטֵר?" וְהַמַּהַרַ"ל אָמַר לָהֶם: "לְכוּ לַכְּפָר הַקָּטָן, וְתִשְׁאֲלוּ מִי זֶה: "אֶפְרַיִם עוֹלְלוֹת"? - הוּא יִהְיֶה הָרַב שֶׁלָּכֶם.." וְהַמַּהַרַ"ל נִפְטַר.

הַתַּלְמִידִים הָלְכוּ לַכְּפָר הַקָּטָן וְשָׁאֲלוּ אֶת כֻּלָּם: "אֵיפֹה נִמְצָא אֶפְרַיִם עוֹלְלוֹת?" אֲבָל אַף אֶחָד לֹא יָדַע.. פִּתְאוֹם בָּא מִישֶׁהוּ וְאָמַר: "יֵשׁ אֶפְרַיִם אֶחָד שֶׁהוּא מוֹכֵר בֵּיצִים.. וְתָמִיד הוּא כּוֹתֵב עַל הַדַּפִּים: "אֶפְרַיִם עוֹלְלוֹת".." הַתַּלְמִידִים הָלְכוּ אֵלָיו הַבַּיְתָהּ וְרָאוּ שֶׁהַבַּיִת פָּשׁוּט מְאֹד, כְּמוֹ שֶׁל עֲנִיִּים.. הֵם שָׁאֲלוּ אֶת אִשְׁתּוֹ: "אֵיפֹה הָרַב?" וְהִיא אָמְרָה: "אֵיזֶה רַב? בַּעֲלִי לֹא רַב.. הוּא מוֹכֵר בֵּיצִים.. עוֹד מְעַט הוּא מַגִּיעַ.." אַתֶּם יוֹדְעִים לָמָּה הִיא אָמְרָה כָּךְ?  נָכוֹן, הִיא לֹא רָצְתָה שֶׁיֵּדְעוּ שֶׁזֶּה אֶפְרַיִם שְׁלֹמֹה.. כִּי הָרַב אָמַר לוֹ שֶׁיִּהְיֶה צַדִּיק נִסְתָּר.. הֵם חִכּוּ לוֹ, כְּשֶׁהוּא נִכְנַס אָמְרוּ לוֹ: "רָצִינוּ לוֹמַר לְךָ שֶׁהַמַּהַרַ"ל מִפְּרָאג נִפְטַר.. וְהוּא אָמַר שֶׁאַתָּה תִהְיֶה הָרַב שֶׁלָּנוּ.." שְׁלֹמֹה אֶפְרַיִם בָּכָה עַל הָרַב שֶׁלּוֹ.. וְהִסְכִּים לָבוֹא לִהְיוֹת הָרַב שֶׁלָּהֶם.

יְלָדִים יְקָרִים, רַבִּי שְׁלֹמֹה אֶפְרַיִם כָּתַב סֵפֶר מְיֻחָד שֶׁנִּקְרָא: "עוֹלְלוֹת אֶפְרַיִם", כְּשֶׁתִּהְיוּ גְדוֹלִים תִּלְמְדוּ בּוֹ וְכַמּוּבָן תִּזָּכְרוּ בַּסִּפּוּר הַמְיֻחָד שֶׁהָיָה לוֹ..
