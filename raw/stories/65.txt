סְקְרִיפּטוֹרִיוּם בְּלִי גְּבוּלוֹת אֲנַחְנוּ, הִתְכַּוְּנוּת חָיָה וּבוֹ גּוּפֵנוּ מִתְנוֹדֵד לְקֶצֶב שֶׁל כְּתִיבָה-קְרִיאָה. תְּשׁוּבָה שֶׁנְבַקֵּשׁ נִמְצָא שׁוֹפַעֲת בִּקְפָלִים וְאֲנוּ מִתְקַדְּמִים בָּהֵם גּוּפֵנוּ מִתְפַּעֵם בְּהֹלֶם שֶׁל קְרִיאָה, כְּתִיבָה. שְׁאֵלַתֵּנוּ לֹא פְּשׁוּטָה וְרַק מֻרְכֶּבֶת וְהוֹלֶכֶת הִיא וְכָל שְׁנְפַשְּׁטָה נוֹסִיף לְפִתּוּלֵיָּהּ הַשָּׁבִים כְּלַפֵּי עַצְמָם וְנִקְשָׁרִים בְּלוּלָאוֹת מַכְאוֹב: אַף כִּי מְרוּרִים מְעֻנָּגִים הֵם לָנוּ טַעֲמֵי כְּתִיבָה כִי הֵם תְּשֻׁקָה לַיֵּשׁ הָחַי בִּתְּשׁוּקָתוֹ הִנֵּה, תְּמִיהָה יֶשְׁנָהּ עַל אֲדִישׁוּת הַיֵּש לַסִּפְרִיָּה שֶׁהִיא אֲדִישׁוּתוֹ לְמַעֲשֵׂינוּ שְׁאֵינַם מֻפְנִים נֶגְדּוֹ שְׁכֵּן אֵינוֹ נִפְרָד וּמַעֲשֵׂינוּ, הַקְרִיאָה וְהַכְּתִיבָה הַמְּבַקְּשִׁים אוֹתוֹ, הַרֵי בְכַךְ מֻפְנִים כְּלַפֵּינוּ שֶׁאֲנַחְנוּ הוּא ולֹא-לֹא-יֵשׁ הַסִּפְרִיָּה.
 
רַגְלֵינוּ בּוֹעֲרוֹת בְּגֹּדֶשׁ טַעֲמָן ועֹנֶג אֶצְבְּעוֹת יָדֵינוּ מִתְרַגֵּשׁ בֵּעוֹז מָתוֹק, פּוֹרֵץ, נוֹבֵעַ וּמְרַקֵּד גַּבֵּנוּ בִטְעַמֵיָּהּ שֶׁל תְּשׁוּקָה לְאוֹת, אֲלֵינוּ, אֲל הַיֵּשׁ וְלֹא נֶחְדַּל מֵהִתְבּוֹנֵן, כָּתֹב, חָקֹר, דַּפְדֵף וְהַעֲתֵק אַךְ בּעוֹכְרֵנוּ הַדָּבָר, כִּי יְצִיאָה הַחוּצָה מַשְׁמָעָה אַךְ יְצִיאָה לְכָאן, אֶלֵינוּ וְאֲנַחְנוּ כָאן, אֶצְלֵנוּ, בְּנִבְכֵי הַסִּפְרִיָּה.
 
והַסִפְרִיָּה יֶשְׁנָהּ שָׁוָה הִיא לָאֵינְסוֹף אֲשֶׁר אֵינוֹ שׁוֹנֵה מִמֶנָּה וְנִצֶּבֶת לְעֻמַּת וּמִשְׁתּוֹקֶקֶת רַק אֲלָיו. אַךְ, פַּעַר אֵינְסוֹפִי נִצָּב בֵּינֵינוּ לְבֵינוֹ, זֶה פַעַר שְׁאֶפְשָׁר לִקְרוֹא לוֹ "צֵל הַצָּהֳרַיִם": קַו זָעִיר נִיצָב בֵּין שְׁנַיִם לְלֹא צֵל בְּשִׁיבַתָם, מִצְחָם מַשִּׁיק לַפַּעַר, לְעַמּוּד נִסְתַּר בְּהַר אֵלָיו נוֹשְׂאִים אֲנוּ עֵינֵינוּ, הַר הַיֵּשׁ.
 
בְּמוֹרדוֹתָיו הַאָדִישִׁים חָיָה הַאֲדָמָה עַל מִכְמַנֵּיָהּ, הֶחַיִּים הַצִּמְחִיִים חַיִּים וְהֶחַיִּים, הַכֹּל נֶאֱרַגִים בָּאוֹרְגַנִיזְם הָעָתִיק רָווּי הַנְּשָׁמָה מְלֵא הֶחַיִּים, חַיִּים הַנְּטֻלִים כֹּל סְתָמִיּוּת אוֹ אָדִישׁוּת, חַיִּים שֶׁהֵם רָצוֹן עָשׂוּי מִרְצוֹנוֹת רָבִּים הַמִּתְכַּוְּנִים אֲלֵיהֵם הֶחַיִּים.
 
וְהִנֵּה עַל הָהַר פּוֹרֵץ עַמּוּד שֶׁל אֵשׁ. הִנֵּה צִפּוֹר מַגִּיעַ בְּקוֹלוֹת שֶׁל הֲקָאָה מִרְקִיעַ הַפְּלָדָה וּמְשַׂרְטֵט בָּהַר עִגוּל. הִנֵּה מִסֵתֶר עִרְבּוּבִיָּת גִּרְגּוּר, שִׁקְשׁוּק וְגְּעִיָּה, שְׁרִיקָה, קְרִיאָה וִיְלָלָה וְרָחַשׁ צָב בֵּין הָקוֹצִים, פּוֹרְצִים גִּרְגּוּר, שִׁקְשׁוּק, קְרִיאָה, שְׁרִיקָה וְגְּעִיָּה וְלְחִישָׁה וְזְעָקָה וִיְלָלָה – כֻּלָּם הַיֵּשׁ, כֻּלָּם הַסִּפְרִיָּה כֻלָּהַם פַּעַר בֵּינֵיהֶם שֶׁלְפִיכָךְ אֲנוּ סוֹפְרִים שְׁלֹשָׁה.
 
בָּמַעֲגַל קַו מֵת שֶׁל אֲלֻנְקוֹת וְשְׁאָר הַזְּוָעָה שְׁאֵין לְתָּאַרָהּ אֲשֶׁר הָהַר הַמִּתְנַשֵּׁף אָדִישׁ לָהּ וְהָהַר, מַהוּ לַזֵּה? – זֹאת הִיא הַשְּׁאֵלָה.

תֵּל-אָבִיב נִפְרֶשֶׂת מִשְּׁנֵי צִדָּיו שֶׁל
אִבְּן-גְּבִירוֹל,
זוֹנָה פְּשׂוּקַת רַגְלַיִם, הַמַּזְמִינָה אֶל תּוֹכָהּ
צְעִירִים, לְהַדְבִּיקָם בְּסִרְחוֹן סִמְטְאוֹתֶיהָ.
תֵּל-אָבִיב מְחַלֶּקֶת אֶת עַצְמָהּ לָדַעַת. חֲדָרִים
בְּתוֹךְ דִּירוֹת. דִּירוֹת בְּתוֹךְ חֲדָרִים. אֲנָשִׁים
נִדְחָסִים לְתוֹךְ כּוּכִים, נֶאֱנָקִים
וְנִשְׁאָרִים.
תֵּל-אָבִיב, הָעִיר הַלְּבָנָה, הָעִיר בְּלִי הַפְסָקָה, הָעִיר
שֶׁחֻלְּקָה לָהּ, שֶׁהֶעֶלְתָה לָהּ דְּמֵי שְׂכִירוּת
בְּלִי הַפְסָקָה. דָּמִים בְּלִי חֶשְׁבּוֹן, יָמִים שֶׁל שֵׁנָה,
לֵילוֹת לְלֹא תְּנוּמָה, מַגִּישָׁה חֶשְׁבּוֹן בְּסוֹף כָּל חֹדֶשׁ
וּמֵרֹאשׁ, לַחֹדֶשׁ הַבָּא.
תֵּל-אָבִיב הֶעֶלְתָה אוֹתָךְ לְעוֹלָה. אַתְּ הָיִית
מִנְחָתָהּ. אַתְּ פָּסַעְתְּ הַיְשֵׁר אֶל תּוֹךְ נִיבֶיהָ הַמַּצְחִינִים
וְנִטְרַפְתְּ בְּזֻהֲמָתָהּ. תֵּל-אָבִיב הִנְגִּישָׁה לָךְ מַסְלוּל וְהָרְגָה אוֹתָךְ
בְּדַרְכָּהּ.
הָעִיר הַזֹּו, אָמַרְתְּ, תִּתֵּן לָנוּ מַחֲסֶה מִפְּרוֹבִינְצְיָאלִיּוּת נִבְעֶרֶת,
אֲבָל תִּגְבֶּה אֶת מְחִירָהּ – מְחִיר אֲנָשִׁים בּוֹדְדִים, הַמְּבַקְּשִׁים קִרְבָה מְנַחֶמֶת
בְּעִיר מְנֻכֶּרֶת, לְלֹא הַכָּרָה.
אַתְּ הָיִית
לִיטְרַת הַבָּשָׂר שֶׁלָּהּ, בַּיּוֹם הַהוּא.
