כְּדַאי לָדַעַת, שֶׁבַּגִּיל הַשְּׁלִישִׁי
אֵין רֶגַע דַּל, מֵרִאשׁוֹן עַד שִׁשִּׁי,
לְמַזָּלֵנוּ, הַנְּשִׁימָה קְצָת בְּעָיָתִית
וְהִיא גַּם קְצָת יוֹתֵר אִטִּית,
לָכֵן בֵּין הַנְּשִׁימוֹת אֶפְשָׁר עוֹד לִדְחֹס
וּלְהִשָּׁאֵר בַּחַיִּים בְּטֶרֶם נִקְרֹס.

הלו"ז בְּגִיל זֶה הוּא כָּל כָּךְ עָמוּס
וְצָרִיךְ לִפְעָמִים, לִדְחוֹת אֲנָשִׁים בְּנִימוּס,
לְדֻגְמָה, הִזְמִין אוֹתִי הָרַב הָרָאשִׁי
דָּחִיתִי אוֹתוֹ כִּי קָבַעְתִּי כְּבָר עִם הַנָּשִׂיא,
עִם נְשִׂיא גִּמְלָאֵי פּוֹעֲלֵי הַבִּנְיָן
כִּי הָיָה לָנוּ חָשׁוּב, לִסְגֹּר אֵיזֶה עִנְיָן.
דָּחִיתִי גַּם פְּגִישָׁה עִם נְשִׂיא הַמְּדִינָה
בִּגְלַל הַרְצָאָה חֲשׁוּבָה בְּנוֹשֵׂא הַקְּרִינָה,
בִּכְלָל, אֲנַחְנוּ שׁוֹמְעִים הַרְצָאוֹת בְּכַמּוּת
בַּנּוֹשֵׂא הֶחָשׁוּב: אֵיךְ לִחְיוֹת בְּלִי לָמוּת !
אֲבָל אֹמַר לָכֶם אֶת הָאֱמֶת,
בְּכָל זֹאת, בְּכָל חֹדֶשׁ מִי שֶׁהוּא מֵת.

מַרְצֶה אֶחָד אָמַר:שֶׁבִּגְלַל הָאֹכֶל שֶׁל הַגַּן
יֵשׁ לָנוּ הַיּוֹם, בַּבֶּטֶן בָּלָגָן,
רָאִיתִי בְּתוֹכֵנוּ שֶׁבִּרְצִינוּת מִתְיַחֲסִים
וְהֵם עַל הַגַּנֶּנֶת, מָה זֶה כּוֹעֲסִים.
מַרְצֶה אַחֵר אָמַר: שֶׁבָּרָצוֹן לְזוֹלְלוּת
אָשֵׁם מוֹרֶה הַהִתְעַמְּלוּת.
מָה שֶׁבּוֹלֵֵט אֵצֶל מַרְבִּית הַמַּרְצִים
שֶׁהֵם מֵעַצְמָם, מְאֹד מְרוּצִים.

כְּדַאי לָדַעַת, שֶׁבַּגִּיל הַשְּׁלִישִׁי
אֵין רֶגַע דַּל, מֵרִאשׁוֹן עַד שִׁשִּׁי.

תַּאֲמִינוּ לִי, בַּגִּיל הַשְּׁלִישִׁי מַמָּשׁ חוֹגְגִים
שׁוֹמְעִים הַרְצָאוֹת וּמַמְצִיאִים חוּגִים,
אֶתְמוֹל נִרְשַׁמְתִּי לְחוּג בּוֹ לוֹמְדִים :
אֵיךְ צוֹעֲדִים עַל הַמַּאֲדִים !
זֶה חוּג בִּכְלָל לֹא קַל
כִּי יֵשׁ לְהַפְגִּין, הֲמוֹן שִׁוּוּי מִשְׁקָל.
שָׁמַעְתִּי שֶׁעוֹד חוּג נִפְתַּח אֶתְמוֹל
חוּג לִגְמִילָה מֵאָקָמוֹל,
וְיֶשְׁנָם חוּגִים בָּהֶם יֵשׁ אִינְפְלַצְיָה
אֵלֶּה הַחוּגִים עִם הַמִּלָּה : תֶּרַפְּיָה.

קַיָּם חוּג : מָתַי וְכֵיצַד נָכוֹן לְהַשְׁכִּים ?
וְחוּג נוֹסָף: מָתַי נָכוֹן לֹא לְהַסְכִּים ?

בֵּין עוּגִיָּה לְעוּגִיָּה
שָׁמַעְתִּי הַרְצָאוֹת אַחַת אַחֲרֵי הַשְּׁנִיָּה,
הָרִאשׁוֹנָה, עַל שְׁתִיָּה לְלֹא קֶשֶׁר לְצָמָא
וְהַשְּׁנִיָּה, עַל נִזְקֵי תַּפּוּחַ אֲדָמָה,
וְעוֹד בְּאוֹתָהּ הַיְּמָמָה :
הַאִם בְּגִיל זֶה יֵשׁ תּוֹפָעוֹת שֶׁל זִמָּה ?
וְאֵיךְ תּוֹפְסִים, בַּעַל שֶׁרִמָּה ?

קַיָּם גַּם חוּג וְהוּא רַק לַנָּשִׁים :
אֵיךְ אֶפְשָׁר אֶת הַגֶּבֶר לְהַרְשִׁים ?
וְיֵשׁ לָהֶן חוּג שֶׁהוּא לֹא מְפֻרְסָם :
אֵיךְ יוֹצְרִים, סוּגִים שׁוֹנִים שֶׁל סַם ?

גַּם לַגְּבָרִים יֵשׁ חוּג סוֹדִי :
כֵּיצַד תְּגַלֶּה, שֶׁאַתָּה לֹא בִּלְעָדִי ?

גִּדּוּל נְכָדִים לְלֹא טָעוּת
נִתַּן לִלְמֹד בַּחוּג לְסַבְתָּאוּת.

קוֹרֶה וּבְהַרְצָאוֹת יֵשׁ נִרְדָּמִים
אֲבָל כָּכָה בְּכֵּיף, עוֹבְרִים הַיָּמִים,
וְלִפְעָמִים גַּם מִבֵּין הַשּׁוּרוֹת
שׁוֹמְעִים פֹּה וָשָׁם נְחִירוֹת,
וּבֶאֱמֶת מִי שֶׁהֲכִי עֲיֵפִים אוֹ עֲיֵפוֹת
הֵם הַמְּנַקִּים בַּשֵּׁרוּתִים אֶת הַמִּרְצָפוֹת.

כְּדַאי לָדַעַת, שֶׁבַּגִּיל הַשְּׁלִישִׁי
אֵין רֶגַע דַּל, מֵרִאשׁוֹן עַד שִׁשִּׁי.

עִיַּפְתִּי מֵחוּגִים, מֵהַרְצָאוֹת וּמִמַּטָּלָה לְמַטָּלָה,
הָלַכְתִּי לָרוֹפֵא וּבִקַּשְׁתִי יְמֵי מַחֲלָה,
הוּא נָתַן לִי מִרְשָׁם וּבוֹ הַתְרָאָה:
אַל תְּוַתֵּר עַל שׁוּם הַרְצָאָה !
וְהוּא הוֹסִיף בַּתַּחְתִּית הַמְלָצָה : 
יֵשׁ חוּג שֶׁתּוֹפֵס הַיּוֹם תְּאוּצָה,

חוּג נִפְלָא בְּשֵׁם : "הוּא הָיָה"
חוּג לִבְדִיחוֹת בְּעֵת לְוָיָה.

נִרְשַׁמְתִּי לַחוּג, אַף שֶׁהוּא קְצָת רָחוֹק
אֲבָל יֹפִי שֶׁל חוּג, מֵתִים בּוֹ מִצְּחוֹק.

אֵין רֶגַע דַּל, מֵרִאשׁוֹן עַד שִׁשִּׁי
כֵּיף לִחְיוֹת בַּגִּיל הַשְּׁלִישִׁי !
