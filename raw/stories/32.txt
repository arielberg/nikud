לִפְנֵי שָׁנִים רַבּוֹת הָיָה מֶלֶךְ גָּדוֹל וְעָצוּם וְקָרְאוּ לוֹ 'נְבוּכַדְנֶצַּר' מֶלֶךְ בָּבֶל. הוּא הָיָה רָשָׁע גָּדוֹל, וּבִגְלַל שֶׁהָיוּ לְעַם יִשְׂרָאֵל עֲווֹנוֹת, ה' נָתַן לוֹ רְשׁוּת לְהַחֲרִיב אֶת בֵּית הַמִּקְדָּשׁ הָרִאשׁוֹן, וְלָקַחַת אֶת עַם יִשְׂרָאֵל לַמְּדִינָה שֶׁלּוֹ: בָּבֶל. פַּעַם אַחַת, הֶחְלִיט נְבוּכַדְנֶצַּר לַעֲשׂוֹת פֶּסֶל עֲנָק כְּדֵי שֶׁכֻּלָּם יִשְׁתַּחֲווּ לוֹ, גַּם הַגּוֹיִים וְגַם הַיְּהוּדִים. כָּל הַפֶּסֶל הָיָה מִזָּהָב וְהוּא הָיָה בְּגֹבַהּ שֶׁל כִּמְעַט שְׁלוֹשִׁים מֶטֶר, אַתֶּם יוֹדְעִים אֵיזֶה גֹבַהּ זֶה? זֶה כְּמוֹ גֹּבַהּ שֶׁל בִּנְיָן עִם עֶשֶׂר קוֹמוֹת..

אֶחָד הַחַיָּלִים צָעַק בְּקוֹל גָּדוֹל: "בְּרֶגַע שֶׁיִּשָּׁמַע קוֹל הַשּׁוֹפָר, וְהַמַּשְׁרוֹקִית, וְהַכִּנּוֹר, וְהֶחָלִיל, וְהַתֹּף וְכָל הַתִּזְמֹרֶת הַגְּדוֹלָה, כֻּלָּם צְרִיכִים מִיָּד לְהִשְׁתַּחֲווֹת לַפֶּסֶל הַגָּדוֹל. וּמִי שֶׁלֹּא יִשְׁתַּחֲוֶה, נְבוּכַדְנֶצַּר הָרָשָׁע יִזְרֹק אוֹתוֹ לְתוֹךְ מְדוּרָה שֶׁל אֵשׁ!!" כֻּלָּם פָּחֲדוּ מְאֹד מִנְּבוּכַדְנֶצַּר הָרָשָׁע, וְלָכֵן, בְּרֶגַע שֶׁהֵם שָׁמְעוּ אֶת הַקּוֹל שֶׁל הַשּׁוֹפָר, וְהַמַּשְׁרוֹקִית, וְהֶחָלִיל, וְהַכִּנּוֹר, וְהַתֹּף וְכָל הַתִּזְמֹרֶת, מִיָּד, הִשְׁתַּחֲווּ כֻּלָּם לַפֶּסֶל הַגָּדוֹל שֶׁעָשׂוּי מִזָּהָב וּנְבוּכַדְנֶצַּר יָשַׁב וְרָאָה אֶת זֶה וְשָׂמַח מְאֹד. לְפֶתַע, נִכְנְסוּ כַּמָּה אֲנָשִׁים לְתוֹךְ הָאַרְמוֹן וְאָמְרוּ: "נְבוּכַדְנֶצַּר, נְבוּכַדְנֶצַּר, אַתָּה חַיָּב לִשְׁמוֹעַ מָה קָרָה.." "כֵּן, מָה קָרָה? תְּסַפְּרוּ לִי.." וְהֵם אָמְרוּ לוֹ: "אֲנַחְנוּ הִסְתַּכַּלְנוּ לִרְאוֹת שֶׁבֶּאֱמֶת כָּל הָאֲנָשִׁים מִשְׁתַּחֲוִים לַפֶּסֶל, וְרָאִינוּ שֶׁהָיוּ שְׁלוֹשָׁה אֲנָשִׁים יְהוּדִים שֶׁלֹּא הִשְׁתַּחֲווּ בִּזְמַן שֶׁהַשּׁוֹפָר וְהַמַּשְׁרוֹקִית וְכָל הַתִּזְמֹרֶת נִגְּנָה מָה"??"

צָעַק נְבוּכַדְנֶצַּר, "מִי הֵם שְׁלוֹשֶׁת הָאֲנָשִׁים הַחֲצוּפִים הַלָּלוּ?" הֵם אָמְרוּ לוֹ: "קוֹרְאִים לָהֶם: חֲנַנְיָה, מִישָׁאֵל וַעֲזַרְיָה. הֵם שְׁלוֹשָׁה יְהוּדִים שֶׁאַתָּה נָתַתָּ לָהֶם תַּפְקִידִים חֲשׁוּבִים בְּבָבֶל וְהֵם לֹא מְקַיְּמִים אֶת הַפְּקֻדָּה שֶׁלְּךָ.." נְבוּכַדְנֶצַּר כָּעַס נוֹרָא וְאָמַר: "הֲבִיאוּ אוֹתָם אֵלַי מִיָּד!!" הַחַיָּלִים הָלְכוּ וְהֵבִיאוּ אֶת שְׁלֹשְׁתָּם לַמֶּלֶךְ. נְבוּכַדְנֶצַּר הִסְתַּכֵּל עֲלֵיהֶם וְאָמַר: "אֵיךְ הֵעַזְתֶּם לֹא לְהִשְׁתַּחֲווֹת לַפֶּסֶל? הֲרֵי אֲנִי אָמַרְתִּי שֶׁמִּי שֶׁלֹּא יִשְׁתַּחֲוֶה אֶזְרֹק אוֹתוֹ לְתוֹךְ כִּבְשַׁן הָאֵשׁ!!". יְלָדִים, אַתֶּם יוֹדְעִים מָה זֶה 'כִּבְשַׁן אֵשׁ'? נַסְבִּיר לָכֶם, חוֹפְרִים בּוֹר גָּדוֹל מְאֹד, וּבְתוֹכוֹ זוֹרְקִים חֹמָרִים דְּלִיקִים שֶׁמַּבְעִירִים אֶת הָאֵשׁ וְהָאֵשׁ בּוֹעֶרֶת בִּפְנִים בְּחֹם גָּבוֹהַּ מְאֹד וְזֶה נִקְרָא 'כִּבְשַׁן אֵשׁ'. נְבוּכַדְנֶצַּר הִמְשִׁיךְ וְאָמַר לָהֶם: "אֲנִי נוֹתֵן לָכֶם הִזְדַּמְּנוּת אַחֲרוֹנָה, עוֹד מְעַט יִתְקְעוּ בַּשּׁוֹפָר, בַמַּשְׁרוֹקִית, בַכִּנּוֹר וּבַתֹּף וְאַתֶּם תִּשְׁתַּחֲווּ, וְאִם לֹא - אַעֲנִישׁ אֶתְכֶם..

חֲנַנְיָה, מִישָׁאֵל וַעֲזַרְיָה לֹא פָּחֲדוּ, אֶלָּא אָמְרוּ לוֹ: "אֲנַחְנוּ לֹא צְרִיכִים עוֹד הִזְדַּמְּנוּת, כִּי אֲנַחְנוּ יְכוֹלִים לוֹמַר לְךָ מֵעַכְשָׁו שֶׁאֲנַחְנוּ לֹ-א נִ-שְׁ-תַּ-חֲ-וֶּ-ה לַפֶּסֶל שֶׁלְּךָ!!" נְבוּכַדְנֶצַּר נִהְיָה אָדֹם כְּמוֹ סֶלֶק אָדֹם, וְהוּא צָעַק: "מָה?? לָמָּה??" וְהֵם אָמְרוּ לוֹ: "כִּי אֲנַחְנוּ עוֹבְדִים אֶת בּוֹרֵא הָעוֹלָם וְלֹא אֶת הַפֶּסֶל, וְה' יָכוֹל לְהַצִּיל אוֹתָנוּ מִכָּל צָרָה וְגַם מִכִּבְשַׁן הַאֵשׁ שֶׁהֵכַנְתָּ, וְגַם אִם ה' לֹא יִרְצֶה לְהַצִּיל אוֹתָנוּ, בְּכָל מִקְרֶה, אֲנַחְנוּ מוּכָנִים לָמוּת וְלֹא לְהִשְׁתַּחֲווֹת לַפֶּסֶל!!" יְלָדִים יְקָרִים, אַתֶּם צְרִיכִים לָדַעַת שֶׁבְּכָל שְׁאַר הָעֲבֵרוֹת, מֻתָּר לַעֲבוֹר עֲבֵרָה כְּדֵי לְהַצִּיל אֶת הַחַיִּים שֶׁלָּנוּ, לְדֻגְמָא: מֻתָּר לְחַלֵּל שַׁבָּת בִּשְׁבִיל חוֹלֶה שֶׁיֵּשׁ סַכָּנָה לְחַיָּיו. דֻּגְמָא אַחֶרֶת: מֻתָּר לְחוֹלֶה מְסֻכָּן לֶאֱכוֹל בְּיוֹם כִּפּוּר, אֲבָל אָסוּר לָאָדָם לַעֲבוֹד עֲבוֹדָה זָרָה כְּגוֹן: לְהִשְׁתַּחֲווֹת לַפֶּסֶל, אֲפִילוּ אִם יַהַרְגוּ אוֹתוֹ אִם לֹא יַעֲבֹד אֶת הָעֲבוֹדָה זָרָה. וְלָכֵן, חֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה הָיוּ מוּכָנִים לָמוּת. נְבוּכַדְנֶצַּר אָמַר לָהֶם: "אִם כָּךְ, אֲנִי אַעֲנִישׁ אֶתְכֶם כְּמוֹ שֶׁאָמַרְתִּי.." הוּא צִוָּה לָשִׂים בְּתוֹךְ הַתַּנּוּר נֵפְט, זֶפֶת, עֵצִים וְעוֹד הֲמוֹן חוֹמָרִים שֶׁבּוֹעֲרִים בָּאֵשׁ, הַכִּבְשָׁן הִתְחַמֵּם מְאֹד מְאֹד עַד שֶׁהַלְּהָבוֹת שֶׁל הָאֵשׁ עָלוּ לְגֹבַהּ שֶׁל יוֹתֵר מֵעֶשְׂרִים מֶטֶר.. שְׁלוֹשֶׁת הַצַּדִּיקִים: חֲנַנְיָה, מִישָׁאֵל וַעֲזַרְיָה לָבְשׁוּ אֶת בִּגְדֵי הַשַּׁבָּת שֶׁלָּהֶם, כִּי הֵם רָצוּ לְקַיֵּם מִצְוָה עִם בְּגָדִים יָפִים.. אֵיזוֹ מִצְוָה? מִצְוַת "קִדּוּשׁ ה'". וְהָלְכוּ בְּשִׂמְחָה לְכִוּוּן הַכִּבְשָׁן. הַחַיָּלִים קָשְׁרוּ אוֹתָם בַּיָּדַיִם וּבָרַגְלַיִם, וּבְבַת אַחַת זָ-רְ-ק-וּ אוֹתָם לְתוֹךְ הַכִּבְשָׁן..

אוֹתָם הַחַיָּלִים, שֶׁהִתְקָרְבוּ לָאֵשׁ כְּדֵי לִזְרֹק אוֹתָם, יָצְאוּ לְהָבוֹת חֲזָקוֹת מֵהַכִּבְשָׁן וְנִשְׂרְפוּ וּמֵתוּ. כֵּן, זֶה הָיָה מְאֹד מְאֹד חַם.. אֲבָל פִּתְאוֹם קָרָה מַשֶּׁהוּ מוּזָר.. מָה קָרָה? כִּבְשַׁן הָאֵשׁ שֶׁהָיָה בְּתוֹךְ הַבּוֹר בָּאֲדָמָה הִתְרוֹמֵם לְמַעְלָה.. כָּל הָאֲנָשִׁים בָּרְחוּ לְמֵרָחוֹק וְהִסְתַּכְּלוּ עָלָיו.. וּוּאַי.. זֶה לֹא יְאֻמַּן!! הֵם רָצוּ לִנְבוּכַדְנֶצַּר וְקָרְאוּ לוֹ: "בֹּא, בֹּא תִרְאֶה מָה קוֹרֶה כָּאן.." נְבוּכַדְנֶצַּר רָץ מַהֵר וּמָה הוּא רָאָה יַחַד אִתָּם?.. מִחוּץ לַכִּבְשָׁן הָיָה חַם מְאֹד, אֲבָל בְּתוֹךְ הַכִּבְשָׁן נִהְיָה קַר.. חֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה הָיוּ הוֹלְכִים בְּתוֹךְ הַכִּבְשָׁן כְּשֶׁיֵּשׁ אִתָּם עוֹד מִישֶׁהוּ.. "הֵי, מָה זֶה? אַרְבָּעָה אֲנָשִׁים? מֵאֵיפֹה הִגִּיעַ הָאִישׁ הָרְבִיעִי?.. וְאֵיךְ הֵם לֹא נִשְׂרָפִים?.." נְבוּכַדְנֶצַּר הִסְתַּכֵּל וְאָמַר: "אֲנִי מַכִּיר אֶת הָאִישׁ הָרְבִיעִי.. הוּא לֹא אָדָם אֶלָּא מַלְאָךְ!! הוּא הַמַּלְאָךְ שֶׁהָרַג אֶת כָּל הַצָּבָא שֶׁל סַנְחֵרִיב מָתַי שֶׁהָיִינוּ לְיַד יְרוּשָׁלַיִם.." כֵּן יְלָדִים, זֶה הָיָה הַמַּלְאָךְ גַּבְרִיאֵל.. עַכְשָׁו ה' שָׁלַח אוֹתוֹ לְקָרֵר אֶת הַכִּבְשָׁן וּלְהַצִּיל אֶת שְׁלוֹשֶׁת הַצַּדִּיקִים.. מִיָּד קָרָא נְבוּכַדְנֶצַּר לְחֲנַנְיָה, מִישָׁאֵל וַעֲזַרְיָה.. "צְאוּ בְּבַקָּשָׁה מִתּוֹךְ הַכִּבְשָׁן.." הֵם יָצְאוּ מֵהָאֵשׁ, נְבוּכַדְנֶצַּר וְכָל הַשָּׂרִים רָאוּ שֶׁלֹּא קָרָה לָהֶם כְּלוּם. הַגּוּף שֶׁלָּהֶם לֹא נִשְׂרַף, גַּם הַשֵּׂעָר שֶׁבְּדֶרֶךְ כְּלָל נִשְׂרַף מַהֵר מֵהָאֵשׁ - לֹא נִשְׂרַף, וַאֲפִילוּ רֵיחַ שֶׁל אֵשׁ לֹא הָיָה לָהֶם בַּבְּגָדִים.. מִיָּד אָמַר נְבוּכַדְנֶצַּר: "בָּרוּךְ ה' הָאֱלֹקִים שֶׁל הַיְּהוּדִים, שֶׁהִצִּיל אֶת הָעֲבָדִים שֶׁלּוֹ שֶׁלֹּא הִשְׁתַּחֲווּ לַפֶּסֶל", מִיַּד צִוָּה נְבוּכַדְנֶצַּר וְאָמַר: "כָּל מִי שֶׁיְּדַבֵּר לֹא יָפֶה עַל הָאֱלֹקִים שֶׁל הַיְּהוּדִים - יֵעָנֵשׁ מִמֶּנִּי עֹנֶשׁ חֲמוּר בְּיוֹתֵר!" הוּא נָתַן לַחֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה תַּפְקִידִים חֲשׁוּבִים בַּמְּדִינָה שֶׁלּוֹ, וְגַם שָׁלַח מִכְתָּב לְכָל הָאֲנָשִׁים בַּמַּמְלָכָה שֶׁלּוֹ, וְשָׁם הוּא שִׁבֵּחַ אֶת ה' מְאֹד עַל הַנִּסִּים הַגְּדוֹלִים שֶׁעָשָׂה.

יְלָדִים יְקָרִים, רְאִיתֶם אֵיזֶה אֱלֹקִים גָּדוֹל יֵשׁ לָנוּ? אֲפִילוּ הַגּוֹיִים רוֹאִים בַּסּוֹף כַּמָּה שֶׁהוּא גָדוֹל. וְעוֹד מְעַט, כְּשֶׁיָּבוֹא הַמָּשִׁיחַ כּוּוּוּלָם יִרְאוּ שׁוּב שֶׁה' הוּא הַמֶּלֶךְ.
