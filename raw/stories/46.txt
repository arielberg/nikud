לֹא פַּעַם שָׁאַלְתִּי אֶת עַצְמִי: אַבָּא, רִבּוֹן הָעוֹלָמִים, מָה הַתִּקּוּן שֶׁלִּי? מָה אֲנִי מְתַקֶּנֶת בְּזֶה הָעוֹלָם? וְלֹא פַּעַם פָּגַעְתִּי בַּהוֹרִים וּבַאֲחֵרִים, כִּי הֵבַנְתִּי שֶׁאֲנִי צְרִיכָה לִלְמֹד לִדְבֹּק תָּמִיד בַּבּוֹרֵא יִתְבָּרַךְ.
לֹא פַּעַם אֲנִי עֲסוּקָה בִּשְׁאֵלֹות רַבּוֹת, בְּכָל הַתְּשׁוּבוֹת רַק לִהְיוֹת בִּדְבֵקוּת תָּמִיד בְּצַדִּיק אֱמֶת. הַיּוֹם אֲנִי עֲסוּקָה בִּלְהָבִין אֶת תַּכְלִית הַמִּשְׁכָּן שֶׁבָּנִיתִי בְּתוֹכִי.
לֹא פַּעַם נָפַלְתִּי וְקַמְתִּי, וּמִיָּד אָמַרְתִּי: תַּקְטִינִי אֶת עַצְמֵךְ כְּנֶגֶד מַדְרֵגַת עַצְמֵךְ.
לֹא פַּעַם הִדְלַקְתִּי נֵר לַצַּדִּיק, וְכָךְ הִרְגַשְׁתִּי לִהְיוֹת נִבְדֶּלֶת וּמַשְׁפִּיעָה, לִהְיוֹת מְחֻבֶּרֶת לְאוֹר הַבּוֹרֵא, לִהְיוֹת דְּבוּקָה בַּבּוֹרֵא. הַיּוֹם אֲנִי מְקַשֶּׁרֶת אֶת עַצְמִי לְכָל הַצַּדִּיקִים הָאֲמִתִּיִים שֶׁבְּדוֹרֵנוּ וּלְכָל הַצַּדִּיקִים שׁוֹכְנֵי עָפָר.
לֹא פַּעַם בָּכִיתִי בְּלֵב נִשְׁבָּר וְאָמַרְתִּי: אַבָּא, אַתָּה אוֹהֵב אוֹתִי וַאֲנִי רוֹצָה לְהִתְקָרֵב יוֹתֵר חָזָק, לִשְׁמֹר עַל טָהֳרָה וְלִהְיוֹת קְדוֹשָׁה.
לֹא פַּעַם לֹא אָהַבְתִּי אֶת עַצְמִי וּבִקַּשְׁתִּי מִמְּךָ שֶׁאֶזְכֶּה לְצַחְצֵחַ אֶת עַצְמִי, כֵּן, לִהְיוֹת צִנּוֹר שֶׁל הַשְׁפָּעָה, לְקַיֵּם בְּתוֹכִי אֶת הָרָצוֹן לִהְיוֹת בַּתַּכְלִית.
לֹא פַּעַם בָּרַחְתִּי מֵהַבְּשׂוֹרָה כִּי הֵבַנְתִּי שֶׁהַבְּשׂוֹרָה תָּבוֹא, וְיָדַעְתִּי לוֹמַר לַבְּשׂוֹרָה: הַיּוֹם אַתְּ מוּכָנָה, וְכָךְ הִיא אָמְרָה לִי: אַתְּ דְּבוּקָה בַּצַּדִּיק, וְכָל חַיַּיִךְ זֶה לִהְיוֹת דְּבוּקָה בְּצַדִּיקֵי אֱמֶת וּבַשֵּׂם יִתְבָּרַךְ.
לֹא פַּעַם הִטַּלְתִּי סָפֵק בַּאֲמִתּוּתָם שֶׁל דְּבָרִים, וּמִיָּד הֵבַנְתִּי שֶׁרַק הַתּוֹרָה הִיא אֱמֶת, אָז הָלַכְתִּי בְּהַצָּעַת מוֹרִי לִלְמֹד שֶׁהַתּוֹרָה נוֹתֶנֶת לָנוּ לִשְׁמֹר עַל טָהֳרַת הַמַּחֲשָׁבָה.
לֹא פַּעַם בִּקַּשְׁתִּי לְהִתְכַּלֵּל בַּהִתְכַּלְּלוּת שֶׁל הַצַּדִּיקִים, וְהֵבַנְתִּי הַנִּסְּיוֹנוֹת וְנִלְחַמְתִּי בָּהֶם. לֹא פַּעַם בִּקַּשְׁתִּי שֶׁתִּכָּנֵס הַשִּׂמְחָה לְתוֹךְ לִבִּי, וְהָעִקָּר לְהַצְלִיחַ לְפַנּוֹת לָהּ מָקוֹם, הַכֹּל עִנְיָן שֶׁל עֲבוֹדָה פְּנִימִית, שֶׁאֲנִי צְרִיכָה לְהַקְטִין אֶת עַצְמִי, שֶׁעֲנָוָה גְּדוֹלָה מִכֻּלָּם וְשֶׁאֲנִי אֶלְמַד לְהִתְקָרֵב וְלִלְמֹד עַל אֵלּוּ הַמַּתָּנוֹת שֶׁזִּכָּנִי הַבּוֹרֵא שֶׁקַּיָּמוֹת בְּתוֹכִי.
לֹא פַּעַם נָשָׂאתִי תְּפִלָּה וְיָדַעְתִּי שֶׁתְּפִלָּתִי חֲשׁוּבָה לִפְנֵי מֶלֶךְ חַי וְקַיָּם. לֹא פַּעַם שָׁכַחְתִּי לְהוֹדוֹת וּמִיָּד אָמַרְתִּי: שְׁמַע יִשְׂרָאֵל ה’ אֱלֹהֵינוּ ה’ אֶחָד. לֹא פַּעַם הָרַסְתִּי אֶת עַצְמִי עַל יְדֵי הַמַּחֲשָׁבָה וּמִיָּד בָּנִיתִי אֶת מִשְׁכָּנִי הַפְּנִימִי עַל יְדֵי שִׁוִּיתִי, רִבּוֹנוֹ שֶׁל עוֹלָם, שִׁוִּיתִי תָּמִיד לְנֶגְדִּי. לֹא פַּעַם רָאִיתִי נִסִּים בְּדַרְכִּי וּמִיָּד אָמַרְתִּי: כִּי לְעוֹלָם חַסְדּוֹ.
לֹא פַּעַם שָׁכַחְתִּי לֶאֱהֹב, וּמִיָּד חָזַרְתִּי לֶאֱהֹב אַהֲבָה אַחַת שֶׁלֹּא מְדַבֶּרֶת בְּמִלִּים רַק שׁוֹלַחַת סִימָנִים, אַהֲבַת הַבּוֹרֵא שֶׁהִיא הַכֹּחַ שֶׁלִּי בְּזֶה הָעוֹלָם.
לֹא פַּעַם הִתְגַּעְגַּעְתִּי לַבַּעַל שֵׁם טוֹב הַקָּדוֹשׁ וּמִיָּד נָסַעְתִּי לֵילֵךְ לַצַּדִּיק וְאָמַרְתִּי: גַּעְגּוּעַי לַצַּדִּיק. שֶׁבָּאתִי עַל צִיּוּנוֹ אָמַרְתִּי: קוּמָה נָא אוֹר שִׁבְעַת הַיָּמִים, כִּי הַפַּעַם אֲנִי יוֹדַעַת לְהוֹדוֹת, הַפַּעַם אֲנִי שׁוֹנָה, כִּי הַפַּעַם הִלְבַּשְׁתִּי אֶת סִפּוּרֶיךָ וְהִפְעַמְתִּי אֶת נִגּוּנֶיךָ בְּכָל אַהֲבָתִי שֶׁנִּכְסְפָה לְמֶלֶךְ חַי, הַפַּעַם וְכָל פַּעַם אֲנִי יוֹדַעַת שֶׁמּוֹלִיכִים אוֹתִי מִשָּׁמַיִם, אָמֵן וְאָמֵן!
