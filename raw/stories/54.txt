שְׁמִי אֱלִיעֶזֶר.
אֲנִי בֶּן 11 וּמִתְגּוֹרֵר בִּירוּשָׁלַיִם.
בְּלַ"ג בָּעֹמֶר אֲנִי הוּא זֶה שֶׁהִדְלַקְתִּי אֶת הַמְּדוּרָה הַגְּדוֹלָה וְהַמֶּרְכָּזִית שֶׁל לַ"ג בָּעֹמֶר. זוֹ מְדוּרָה שֶׁל מֵאוֹת מִיַּלְדֵי הַשְּׁכוּנָה. חֵלֶק מֵהֶם לֹא יָדְעוּ מַדּוּעַ זָכִיתִי בַּתַּפְקִיד הַמְכֻבָּד הַזֶּה, וְחֵלֶק מֵהֶם חָשְׁבוּ שֶׁהֵם יוֹדְעִים, אֲבָל הֵם טָעוּ לְגַמְרֵי, כִּי לֹא הָיָה לָהֶם מֻשָּׂג מַה קָּרָה שָׁם בֶּאֱמֶת.
רַק שְׁנֵי יְלָדִים יָדְעוּ מַה קָּרָה. אֲנִי וְהָמָן.
כְּלוֹמַר, לֹא בֶּאֱמֶת קוֹרְאִים לוֹ הָמָן, וְכַיּוֹם גַּם אֲנִי לֹא קוֹרֵא לוֹ הָמָן. הַשֵּׁם שֶׁלּוֹ הוּא דַּוְקָא מָרְדְּכַי, אֲבָל בְּמֶשֶׁךְ תְּקוּפָה מְמֻשֶּׁכֶת מְאֹד קָרָאתִי לוֹ הָמָן.
אוּלַי זֶה לֹא בְּסֵדֶר לְכַנּוֹת כָּכָה יֶלֶד, אֲפִלּוּ שֶׁלֹא כִּנִּיתִי אוֹתוֹ בִּפְנֵי אֲחֵרִים, רַק בְּלִבִּי פְּנִימָה כִּנִּיתִי אוֹתוֹ כָּךְ, אֲבָל אִם תֵּדְעוּ כַּמָּה סָבַלְתִּי מִמֶּנּוּ – אוּלַי תָּבִינוּ אוֹתִי, גַּם אִם לֹא תַּצְדִּיקוּ.
אֲנַחְנוּ לוֹמְדִים בְּכִתָּה אַחַת. אֲנִי נִרְאֶה כְּלַפֵּי חוּץ יֶלֶד דֵּי שָׁקֵט וּמֻפְנָם, אֲבָל בִּפְנִים אֲנִי סוֹעֵר מְאֹד וּמָלֵא תָּכְנִיּוֹת וְרַעְיוֹנוֹת.
אֲנִי חוֹשֵׁב שֶׁיֵּשׁ עוֹד יְלָדִים כָּמוֹנִי שֶׁאוּלַי קוֹרְאִים כָּעֵת אֶת הַסִּפּוּר וְיִזְדַּהוּ עִם הַתְּחוּשׁוֹת שֶׁלִּי. מְדֻבָּר בִּילָדִים עִם הֲמוֹן יָזְמָה וְרַעְיוֹנוֹת מְקוֹרִיִּים, שֶׁיֵּשׁ לָהֶם עוֹלָם שֶׁל בְּדִיחוֹת מַצְחִיקוֹת אוֹ שֶׁל דִּבְרֵי חָכְמָה, אַךְ שֶׁכְּשֶׁהֵם רוֹצִים לְהוֹצִיא אֶת זֶה הַחוּצָה – זֶה נִתְקָע, לֹא בָּרוּר מַדּוּעַ. אוּלַי בִּגְלַל חֹסֶר בִּטָּחוֹן עַצְמִי, וְאוּלַי בִּגְלַל בְּעָיָה קְטַנָּה שֶׁל תִּקְשֹׁרֶת.
אֲנִי לֹא אוֹמֵר אֶת זֶה מֵעַצְמִי. קָרָאתִי פַּעַם בָּעִתּוֹן כַּתָּבָה עַל יְלָדִים מֻפְנָמִים, וְהִזְדַּהֵיתִי עִם כָּל מִלָּה. כָּתְבוּ שָׁם עַל יְלָדִים כָּמוֹנִי שֶׁהֵם לֹא מֻפְנָמִים בֶּאֱמֶת, רַק שֶׁיֵּשׁ לָהֶם בְּעָיָה שֶׁל תִּקְשֹׁרֶת – כָּכָה כִּנּוּ אֶת זֶה בַּכַּתָּבָה, וְהֵם מִתְקַשִּׁים לְהוֹצִיא הַחוּצָה אֶת מַה שֶּׁהֵם מַרְגִּישִׁים, חוֹשְׁבִים וְיוֹדְעִים.
הָיוּ שָׁם כָּל מִינֵי מִלִּים מִקְצוֹעִיּוֹת, כְּמוֹ לְמָשָׁל: שִׁיּוּם – קֹשִׁי לִמְצֹא אֶת הַשֵּׁם שֶׁל הַמִּלָּה שֶׁאַתָּה חוֹשֵׁב עָלֶיהָ, וּשְׁלִיפָה – קֹשִׁי לִשְׁלֹף אֶת הַמִּלָּה מִתּוֹךְ הַפֶּה שֶׁלְּךָ.
הַכַּתָּבָה הַזּוֹ הִרְגִּיעָה אוֹתִי מְאֹד, כִּי הִיא עָזְרָה לִי לְהָבִין מַה קּוֹרֶה לִי וְלִילָדִים אֲחֵרִים שֶׁמִּתְקַשִּׁים לְבַטֵּא אֶת עַצְמָם. אֲנִי לֹא יֶלֶד לֹא מְעַנְיֵן, וְלֹא חֲסֵרִים לִי רַעְיוֹנוֹת – פָּשׁוּט קָשֶׁה לִי לְהוֹצִיא הַחוּצָה אֶת מַה שֶּׁבְּתוֹכִי.
אוּלַי נִרְגַּעְתִּי, אֲבָל הַבְּעָיָה נִשְׁאֲרָה וּמְלַוָּה אוֹתִי בְּמֶשֶׁךְ הַשָּׁנִים בַּלִּמּוּדִים וּבַחֶבְרָה.
בַּלִּמּוּדִים אֲנִי מוֹצֵא אֶת עַצְמִי יוֹדֵעַ מַה לַּעֲנוֹת עַל שְׁאֵלָה שֶׁהַמּוֹרֶה שׁוֹאֵל, אֲבָל לִפְנֵי שֶׁאֲנִי מַסְפִּיק לִתְפֹּס אֹמֶץ וּלְדַבֵּר, מִישֶׁהוּ אַחֵר כְּבָר עָנָה עָלֶיהָ, וּבַחֶבְרָה – זֶה כְּבָר סִפּוּר כּוֹאֵב.
מִתְבָּרֵר שֶׁאֵת הַחֶבְרָה מְנַהֲלִים יְלָדִים שֶׁיּוֹדְעִים לְדַבֵּר בְּקוֹלָנִיּוּת וּבִמְהִירוּת, וּבְעִקָּר בְּבִטָּחוֹן עַצְמִי – וּבְדִיּוּק כָּל הַדְּבָרִים הָאֵלֶּה חֲסֵרִים לִי.
יֵשׁ יְלָדִים שְׁקֵטִים שֶׁטּוֹב לָהֶם עִם הַשֶּׁקֶט הַזֶּה, וְהֵם מְקַבְּלִים אֶת הָעֻבְדָּה הַזּוֹ כְּמוּבֶנֶת מֵאֵלֶיהָ. אִם תִּשְׁאַל אוֹתָם אִם הֵם רוֹצִים לִהְיוֹת מַנְהִיגִים אוֹ מַחְלִיטִים, הֵם יֹאמְרוּ לְךָ "לֹא. מַמָּשׁ לֹא".
אֲבָל יֵשׁ מִעוּט שֶׁל יְלָדִים, כָּמוֹנִי לְמָשָׁל, שֶׁדַּוְקָא הָיוּ רוֹצִים מְאֹד לִהְיוֹת בְּמֶרְכַּז הַחֶבְרָה, כִּי הֵם מַרְגִּישִׁים שֶׁיֵּשׁ לָהֶם לָתֵת לֹא פָּחוֹת מֵהָאֲחֵרִים – אֲבָל הַמִּגְבָּלָה הַזּוֹ שֶׁל תִּקְשֹׁרֶת וּבִטָּחוֹן עַצְמִי מַפְרִיעָה לָהֶם.
אֲנִי בָּטוּחַ שֶׁיְּלָדִים רַבִּים שֶׁיִּקְרְאוּ אֶת הַסִּפּוּר הַזֶּה יִזְדַּהוּ עִם כָּל מִלָּה. אֲנִי לֹא בָּטוּחַ שֶׁיֵּשׁ לִי עֵצָה בִּשְׁבִילָם. אוּלַי אַחַת: לְעוֹלָם אַל תִּתְיָאֲשׁוּ, וּמִצְאוּ אֶת הַדֶּרֶךְ לְבַטֵּא אֶת עַצְמְכֶם גַּם בְּלִי לְפַטְפֵּט וְלִצְעֹק. כַּאֲשֶׁר מִתְבַּגְּרִים, כָּךְ אָבִי אָמַר לִי, הַחֶבְרָה מֵעֲרִיכָה יוֹתֵר אֶת הַתֹּכֶן מֵאֲשֶׁר אֶת הַהַצָּגָה, וְדַוְקָא הַקּוֹלָנִיִּים שֶׁמְּדַבְּרִים בְּלִי הַפְסָקָה מְאַבְּדִים אֶת הַקֶּסֶם שֶׁלָהֶם, וְאֵלֶּה שֶׁיֵּשׁ בָּהֶם תֹּכֶן וּמְדַבְּרִים בְּצוּרָה מְתוּנָה תּוֹפְסִים אֶת מְקוֹמָם.
וְכָעֵת לַסִּפּוּר שֶׁרָצִיתִי לְסַפֵּר.
בַּכִּתָּה שֶׁלִּי הָיְתָה קְבוּצָה מְגֻבֶּשֶׁת לְמַדַּי שֶׁל אַרְבָּעָה יְלָדִים שַׁלִּיטִים וְעוֹד כַּמָּה יְלָדִים שֶׁ"הִסְתַּפְּחוּ" אֲלֵיהֶם. הַ"מִּסְתַּפְּחִים" הָיוּ יְלָדִים שֶׁהִתְחַנְּפוּ לַ"שַּׁלִּיטִים" כְּדֵי שֶׁיּוֹאִילוּ בְּטוּבָם לְשַׁתֵּף אוֹתָם מִפַּעַם לְפַעַם. מוּבָן שֶׁהַמִּסְתַּפְּחִים הָיוּ מִתְחַלְּפִים כְּשֶׁסָּר חִנָּם בִּפְנֵי הַשַּׁלִּיטִים.
מַנְהִיג הַשַּׁלִּיטִים הָיָה מָרְדְּכַי, יֶלֶד דּוֹמִינַנְטִי מְאֹד, שֶׁמֵּאָז הַגַּן הָיָה הַמֶּלֶךְ הַבִּלְתִּי מְעֻרְעָר שֶׁל הַכִּתָּה.
וּכְמוֹ בָּעוֹלָם, יֵשׁ מַנְהִיגִים טוֹבִים וְיֵשׁ פָּחוֹת טוֹבִים. יֵשׁ קְשׁוּחִים וְיֵשׁ רַכִּים, יֵשׁ מִתְחַשְּׁבִים וְיֵשׁ אָנֹכִיִּים.
אָז זֶהוּ, שֶׁמָּרְדְּכַי לֹא הָיָה מֵאֵלֶּה הַטּוֹבִים, הָרַכִּים וְהַמִּתְחַשְּׁבִים. הוּא הָיָה שַׁיָּךְ לַסּוּג הָאַחֵר.
וְכָאן צָרִיךְ לְחַלֵּק בֵּין יְלָדִים שְׁקֵטִים בֶּאֱמֶת, שֶׁמָּרְדְּכַי לֹא הָיָה מוֹצֵא בָּהֶם עִנְיָן חוּץ מִכַּמָּה עֶלְבּוֹנוֹת פֹּה וָשָׁם, לְבֵין יְלָדִים כָּמוֹנִי, שֶׁנִּסּוּ לְהַרְאוֹת אֶת עַצְמָם. מָרְדְּכַי הָיָה דּוֹאֵג לְהַשְׁפִּיל אוֹתָם וְלִסְתֹּם לָהֶם אֶת הַפֶּה.
וְאֵיךְ הוּא עָשָׂה זֹאת? בְּעִקָּר בְּכֹחַ הַפֶּה שֶׁלּוֹ. הוּא הָיָה לוֹעֵג לַיְלָדִים, מַשְׁפִּיל אוֹתָם, מְחַקֶּה אוֹתָם וּמֵסִית אֶת הָאֲחֵרִים לִצְחֹק מִדְּבָרִים שֶׁהֵם אוֹמְרִים. אַתֶּם מְבִינִים לְבַד שֶׁהִתְנַהֲגוּת כָּזוֹ פּוֹגַעַת בְּבִטָּחוֹן עַצְמִי גַּם שֶׁל מִי שֶׁיֵּשׁ לוֹ בִּטָּחוֹן עַצְמִי מֻפְרָז, קַל וָחֹמֶר כְּשֶׁמְּדֻבָּר בְּמִישֶׁהוּ שֶׁיֵּשׁ לוֹ בְּעָיָה בַּבִּטָּחוֹן הָעַצְמִי.
עִם הַזְּמַן הִתְחַלְתִּי לִשְׂנֹא אוֹתוֹ מַמָּשׁ. הִרְגַּשְׁתִּי שֶׁהוּא חוֹסֵם אוֹתִי עוֹד יוֹתֵר. פָּחַדְתִּי לְדַבֵּר כְּשֶׁהוּא הָיָה בַּסְּבִיבָה, כִּי הוּא הָיָה מְחַקֶּה אוֹתִי, לוֹעֵג לִי בִּגְלַל מַה שֶּׁאָמַרְתִּי, אוֹ פָּשׁוּט מְבַטֵּל אֶת הַדְּבָרִים בִּתְנוּעָה אַחַת בַּפַּרְצוּף. אַבָּא שֶׁלִּי אָמַר פַּעַם שֶׁכָּתוּב "כִּי בְּאַפָּם הָרְגוּ אִישׁ" – יֵשׁ אֲנָשִׁים שֶׁתְּנוּעָה אַחַת בָּאַף שֶׁלָהֶם יְכוֹלָה לְהַלְבִּין פָּנָיו שֶׁל זוּלָתָם, וְכַיָּדוּעַ, מִי שֶׁמַּלְבִּין פְּנֵי חֲבֵרוֹ הֲרֵי הוּא כְּשׁוֹפֵךְ דָּמִים.
הָיָה זֶה בְּפוּרִים, חֲצִי שָׁעָה לִפְנֵי קְרִיאַת הַמְּגִלָּה, כְּשֶׁנּוֹדַע לִי לְפֶתַע שֶׁחֲצִי מִיַּלְדֵי הַכִּתָּה הִתְאַרְגְּנוּ לִמְסִבָּה בְּשֶׁקֶט בְּשֶׁקֶט. שָׁמַעְתִּי אֶת זֶה מֵחָבֵר, שֶׁסִּפֵּר לִי שֶׁמָּרְדְּכַי הִזְהִיר אֶת כֻּלָּם שֶׁלֹּא לְגַלּוֹת, וּבִפְרָט הִזְהִיר "לֹא לְגַלּוֹת לֶאֱלִיעֶזֶר".
הַמִּלִּים הַלָּלוּ פָּגְעוּ בִּי כְּמוֹ סַכִּין.
יֶלֶד אִרְגֵּן חֲצִי כִּתָּה לְהַסְתִּיר מַשֶּׁהוּ דַּוְקָא מִמֶּנִּי!
לָמָּה הוּא עוֹשֶׂה אֶת זֶה? חָשַׁבְתִּי.
הַלְּחָיַיִם שֶׁלִּי בָּעֲרוּ, וּדְמָעוֹת בִּצְבְּצוּ בְּעֵינַי. לֹא רָצִיתִי שֶׁחֲבֵרִי יִרְאֶה זֹאת, וְלָכֵן הִסְתַּלַּקְתִּי.
בִּקְרִיאַת הַמְּגִלָּה, כְּשֶׁקָּרְאוּ "בָּרוּךְ מָרְדְּכַי", חָשַׁבְתִּי עַל הַמָּרְדְּכַי הַפְּרָטִי שֶׁלִּי. "אֵיזֶה בָּרוּךְ וְאֵיזֶה נַעֲלַיִם", חָשַׁבְתִּי. וְהֶחְלַטְתִּי בְּלִבִּי שֶׁבְּטָעוּת קָרְאוּ לוֹ מָרְדְּכַי. יוֹתֵר מַתְאִים הָיָה לוֹ לְהִקָּרֵא הָמָן.
מֵאוֹתוֹ הַיּוֹם כִּנִּיתִי אוֹתוֹ בְּלִבִּי "הָמָן". יָדַעְתִּי שֶׁיֵּשׁ הֶבְדֵּל גָּדוֹל בֵּינוֹ לְבֵין הָמָן הָאֲמִתִּי, שֶׁרָצָה לְהַשְׁמִיד וְלַהֲרֹג וּלְאַבֵּד אֶת כָּל הַיְּהוּדִים מִנַּעַר וְעַד זָקֵן טַף וְנָשִׁים, אֲבָל אֲנִי רָאִיתִי בּוֹ אֶת "הֶהָמָן הַפְּרָטִי שֶׁלִּי", יֶלֶד שֶׁפּוֹגֵעַ בִּי בְּמֶשֶׁךְ שָׁנִים, חוֹסֵם אֶת דַּרְכִּי בַּחֶבְרָה וְהוֹרֵס אֶת חַיַּי פְּשׁוּטוֹ כְּמַשְׁמָעוֹ.
אֲנִי בָּטוּחַ שֶׁאַתֶּם מַכִּירִים יְלָדִים כָּאֵלֶּה בְּבֵית הַסֵּפֶר שֶׁלָּכֶם אוֹ בַּכִּתָּה שֶׁלָּכֶם. הִנֵּה יֵשׁ לָכֶם הִזְדַּמְּנוּת לְהָבִין מַה הֵם מְעוֹלְלִים.
פּוּרִים חָלַף וְגַם פֶּסַח, וְהִגִּיעָה מְדוּרַת לַ"ג בָּעֹמֶר.
וּכְמוֹ תָּמִיד, הִתְעוֹרְרָה הַשְּׁאֵלָה בְּאֵיזוֹ מְדוּרָה לְהִשְׁתַּתֵּף.
מוּבָן שֶׁמָּרְדְּכַי וַחֲבֵרָיו אִרְגְּנוּ מְדוּרָה. אֲבָל לְעֻמַּת הַשָּׁנִים שֶׁעָבְרוּ, שֶׁבָּהֶן הִתְאַרְגְּנוּ חֲמִשָּׁה יְלָדִים וְעוֹד שְׁנַיִם אוֹ שְׁלֹשָׁה הִסְתַּפְּחוּ אֲלֵיהֶם, הַפַּעַם הֵם הֶחְלִיטוּ לַעֲשׂוֹת אֶת הַמְּדוּרָה לְכָל הַשְּׁכוּנָה.
וְאִם שָׂמַחְתִּי עַל כָּךְ – הַשִּׂמְחָה שֶׁלִּי הָיְתָה מֻקְדֶּמֶת. אָמְנָם כָּל יַלְדֵי הַשְּׁכוּנָה אֲמוּרִים לְהִשְׁתַּתֵּף בַּמְּדוּרָה עַצְמָהּ, אֲבָל זֶה לֹא אוֹמֵר דָּבָר בַּאֲשֶׁר לְאִסּוּף הָעֵצִים וְאִרְגּוּן הַמְּדוּרָה.
מַהֵר מְאֹד הֵבַנְתִּי שֶׁלְּרַבִּים מִיַּלְדֵי הַכִּתָּה חֻלְּקוּ תַּפְקִידִים – וְלִי לֹא.
שׁוּב עָשָׂה מָרְדְּכַי הַכֹּל כְּדֵי לַחְסֹם אוֹתִי. לֹא יוֹדֵעַ לָמָּה. מִסְּבִיבִי רָחֲשׁוּ הַיְלָדִים וּבָחֲשׁוּ בְּנוֹשֵׂא הַמְּדוּרָה, מָה עוֹמֵד לִהְיוֹת וְכַמָּה עֵצִים אָסְפוּ, וַאֲנִי – מִחוּץ לָעִנְיָנִים.
הַהֲמֻלָּה הַגְּדוֹלָה בְּיוֹתֵר הָיְתָה סְבִיב בֻּבַּת הֶ"הָמָן" הָאֲמוּרָה לְהִמָּצֵא בְּמֶרְכַּז הַמְּדוּרָה.
הַהָמָן הָיָה תָּמִיד הָאַטְרַקְצְיָה שֶׁל הַמְּדוּרָה. בְּכָל שָׁנָה הִשְׁתַּדְּלוּ שֶׁהוּא יִהְיֶה גָּדוֹל יוֹתֵר, שָׁמֵן יוֹתֵר וּמַצְחִיק יוֹתֵר. יְלָדִים הָיוּ תּוֹרְמִים בְּגָדִים, כּוֹבָעִים וְנַעֲלַיִם כְּדֵי לְשַׁוּוֹת לַבֻּבָּה מַרְאֶה אֲמִתִּי.
גֹּדֶל הֶהָמָן וְתַהֲלִיךְ בְּנִיָּתוֹ נִשְׁמְרוּ בְּסוֹד, וּכְדֵי שֶׁלֹּא יָבוֹאוּ חַטְטָנִים לְהָצִיץ, גַּם מְקוֹם הִמָּצְאוֹ נִשְׁמַר בְּסוֹד.
וַאֲנִי, כְּפִי שֶׁסִּפַּרְתִּי, מִחוּץ לָעִנְיָנִים.
כְּשָׁבוּעַ לִפְנֵי לַ"ג בָּעֹמֶר הָלַכְתִּי לְיַד בֵּיתוֹ שֶׁל מָרְדְּכַי, וּלְפֶתַע רָאִיתִי אוֹתוֹ בְּרֹאשׁ קְבוּצַת יְלָדִים. הֵם הָלְכוּ לְכִוּוּן מִקְלָט צִבּוּרִי שׁוֹמֵם.
מִיָּד הֵבַנְתִּי שֶׁזֶּהוּ מְקוֹם הַמִּסְתּוֹר שֶׁל הֶ"הָמָן". שָׁם בּוֹנִים אוֹתוֹ, שָׁם מַלְבִּישִׁים אוֹתוֹ וּמְמַלְּאִים אוֹתוֹ בִּסְמַרְטוּטִים, כְּדֵי שֶׁיִּהְיֶה גָּדוֹל וְשָׁמֵן כְּכָל הָאֶפְשָׁר, וְכַמּוּבָן כְּדֵי שֶׁיִּהְיֶה הַרְבֵּה מִמֶּנּוּ לִשְׂרֹף…
מְקוֹם הַמִּסְתּוֹר שֶׁל הֶ"הָמָן" נֶחְשַׁב לְסוֹדִי בְּיוֹתֵר. מוּבָן שֶׁהִסְתַּקְרַנְתִּי לָלֶכֶת וְלִרְאוֹת.
הִמְתַּנְתִּי שָׁם עַד שֶׁמָּרְדְּכַי וַחֲבֵרָיו יִסְתַּלְּקוּ, וּלְאַחַר שֶׁוִּדֵּאתִי שֶׁהֵם נֶעֶלְמוּ מֵהַשֶּׁטַח, הִתְקָרַבְתִּי לַמִּקְלָט.
הָיְתָה שָׁם דֶּלֶת מִבַּרְזֶל שֶׁהֵם סָגְרוּ עַל יְדֵי חוּט בַּרְזֶל.
פָּתַחְתִּי אוֹתָהּ בְּלִי קֹשִׁי וְנִכְנַסְתִּי לַמִּקְלָט.
חֹשֶׁךְ שָׂרַר בַּמָּקוֹם. בְּעַד הַדֶּלֶת הַפְּתוּחָה הִצְלַחְתִּי לִרְאוֹת אֶת הֶהָמָן שׁוֹכֵב שָׁם בְּאֶמְצַע הַמִּקְלָט.
הָיָה זֶה הָמָן בֶּאֱמֶת עֲנָקִי, שֶׁכְּמוֹתוֹ לֹא רָאִיתִי מִיָּמַי. סְבִיבוֹ הָיוּ סְמַרְטוּטִים, מַקְלוֹת וַהֲמוֹן קַשׁ. הַקַּשׁ נוֹעַד לְמִלּוּי הֶהָמָן, כְּדֵי שֶׁיֵּרָאֶה שָׁמֵן.
הִבַּטְתִּי בּוֹ וְחַשְׁתִּי תְּחוּשָׁה מְעֹרֶבֶת: מִצַּד אֶחָד שָׂמַחְתִּי שֶׁגִּלִּיתִי סוֹד שֶׁאַף אֶחָד מִיַּלְדֵי הַכִּתָּה לֹא גִּלָּה חוּץ מֵהַשּׁוֹלְטִים, אַךְ מִצַּד שֵׁנִי חַשְׁתִּי דְּקִירָה חֲזָקָה בְּלִבִּי. יָדַעְתִּי שֶׁאֵין שׁוּם דָּבָר שֶׁמְּקַשֵּׁר בֵּינִי לֶהָמָן הַזֶּה, וּבִכְלָל, בְּכָל הַקָּשׁוּר לַמְּדוּרָה הַזּוֹ נִשְׁאַרְתִּי בַּצַּד. חַשְׁתִּי כְּאֵב בְּלִבִּי. מִי הִרְשָׁה לָהֶם? חָשַׁבְתִּי, בְּאֵיזוֹ זְכוּת הֵם נוֹטְלִים לְעַצְמָם אֶת הַשִּׁלְטוֹן בַּכִּתָּה וְדוֹחֲקִים יְלָדִים אֲחֵרִים, כָּמוֹנִי, לַפִּנָּה?
מַה פֵּרוּשׁ "מְדוּרָה שֶׁלָהֶם"? זוֹ הַכִּתָּה שֶׁלָהֶם?
מַה גּוֹרֵם לַיְלָדִים לָלֶכֶת אַחֲרֵיהֶם? חָשַׁבְתִּי, וּמִיָּד הָיְתָה לִי הַתְּשׁוּבָה:
כָּל הָאֲחֵרִים חוֹשְׁשִׁים פֶּן יַגִּיעוּ לַמַּצָּב שֶׁלִּי. הֵם לֹא רוֹצִים שֶׁיִּדְחֲקוּ אוֹתָם כְּמוֹ שֶׁדָּחֲקוּ אוֹתִי. הֵם מַעֲדִיפִים לְהִתְחַנֵּף וּלְבַטֵּל אֶת עַצְמָם, וּבִלְבַד שֶׁלֹא יַרְחִיקוּ אוֹתָם, וְאִלּוּ אֲנִי לְעוֹלָם אֵינִי מִתְחַנֵּף וּלְעוֹלָם אֵינִי מְבַטֵּל אֶת עַצְמִי בִּפְנֵי אֲחֵרִים. אָמְנָם אֵינֶנִי דַּבְּרָן וְסוֹעֵר כְּמוֹהֶם, אֲבָל בְּתוֹכִי יֵשׁ לִי אִישִׁיּוּת, וְכַנִּרְאֶה הֵם יוֹדְעִים זֹאת וְחוֹשְׁשִׁים לָתֵת לִי אֶפְשָׁרוּת לְהִתְבַּלֵּט.
עָמַדְתִּי בַּמִּקְלָט הֶחָשׁוּךְ, וּפִתְאוֹם חַשְׁתִּי כְּמוֹ יֶלֶד בּוֹדֵד, לְלֹא חֲבֵרִים, מֻרְחָק מִכִּתָּתוֹ וּמִכָּל פְּעִילוּת בָּהּ. אִמָּא שֶׁלִּי קוֹרֵאת לָזֶה "רַחֲמִים עַצְמִיִּים". פָּשׁוּט רִחַמְתִּי עַל עַצְמִי, וְהִתְחַלְתִּי לִבְכּוֹת שָׁם לְבַד, בַּחֹשֶׁךְ. בָּכִיתִי עַל מַר גּוֹרָלִי וְעַל חֹסֶר הַיְכֹלֶת שֶׁלִּי לְשַׁנּוֹת אֶת הַמַּצָּב.
כָּךְ עָמַדְתִּי שָׁעָה אֲרֻכָּה. אֲפִלּוּ לֹא שַׂמְתִּי לֵב כֵּיצַד הַזְּמַן חוֹלֵף.
לְפֶתַע שָׁמַעְתִּי רַעַשׁ מִכִּוּוּן הַדֶּלֶת. קוֹלוֹת שֶׁל יְלָדִים.
הַקּוֹל אָמַר: "מִישֶׁהוּ פָּתַח אֶת דֶּלֶת הַמִּקְלָט".
הָיָה זֶה מָרְדְּכַי.
הֶהָמָן שֶׁלִּי בָּא לְבַקֵּר אֶת הֶהָמָן שֶׁלּוֹ…
הִתְחַבֵּאתִי בְּפִנַּת הַמִּקְלָט. לֹא רָצִיתִי שֶׁהוּא יִקְלֹט שֶׁגִּלִּיתִי אֶת הַסּוֹד הַגָּדוֹל…
הוּא הִגִּיעַ עִם אָחִיו הַקָּטָן. שְׁנֵיהֶם הָיוּ לְבַדָּם.
הֵצַצְתִּי מֵהַפִּנָּה, וְרָאִיתִי שֶׁמָּרְדְּכַי מוֹצִיא נֵר וְגַפְרוּרִים וּמַדְלִיק.
"תַּחְזִיק אֶת הַנֵּר", אָמַר מָרְדְּכַי לְאָחִיו. "בֵּינְתַיִם אֲמַלֵּא אֶת הֶהָמָן בְּקַשׁ".
אָחִיו הֶחְזִיק אֶת הַנֵּר, וּמָרְדְּכַי הֵחֵל לִדְחֹף קַשׁ לְתוֹךְ הַחֻלְצָה שֶׁל "הָמָן".
הִבַּטְתִּי בִּשְׁנֵיהֶם. הַנֵּר הָיָה דֵּי קָטָן. כַּנִּרְאֶה נַעֲשָׂה בּוֹ שִׁמּוּשׁ רַב. מִדֵּי פַּעַם שָׁמַעְתִּי אֶת יָאִיר, אָחִיו שֶׁל מָרְדְּכַי, אוֹמֵר "אַי". הֵבַנְתִּי שֶׁהַשַּׁעֲוָה נוֹטֶפֶת עַל יָדוֹ, וּכְמוֹ שֶׁכֻּלְּכֶם יוֹדְעִים – זֶה כּוֹאֵב.
"אַי!" אָמַר יָאִיר שׁוּב.
עוֹד כַּמָּה 'אַי', וְיָאִיר אָמַר לְמָרְדְּכַי: "תִּשְׁמַע מָרְדְּכַי, הַנֵּר כְּבָר בַּסּוֹף שֶׁלּוֹ, וְכָל הַטִּפּוֹת נוֹפְלוֹת עָלַי. זֶה נוֹרָא כּוֹאֵב, אַיייי!"
"תַּחְזִיק אֶת הַנֵּר וְאַל תְּבַלְבֵּל אֶת הַמֹּחַ", אָמַר מָרְדְּכַי.
בְּדִיּוּק "הֶהָמָן" שֶׁהִכַּרְתִּי.
עוֹד דַּקָּה, וְעוֹד כַּמָּה 'אַי' שֶׁכָּל אֶחָד מֵהֶם חָזָק מִקּוֹדְמוֹ. הִבַּטְתִּי בְּיָאִיר הַמִּסְכֵּן הַמַּחְזִיק אֶת זְנַב הַנֵּר קָרוֹב לְיָדָיו וּמִתְיַסֵּר מִכְּאֵבִים, וּלְמָרְדְּכַי זֶה אֲפִלּוּ לֹא אִכְפַּת.
יָאִיר אוֹמֵר: "מָרְדְּכַי, הַנֵּר קָרוֹב לַיָּדַיִם שֶׁלִּי, זֶה כְּבָר לֹא הַשַּׁעֲוָה! אֲנִי מַרְגִּישׁ שֶׁהָאֵשׁ עוֹשָׂה לִי כְּוִיָּה!"
"בִּקַּשְׁתָּ לָבוֹא, אָז תַּחְזִיק אֶת הַנֵּר וְאַל תְּבַלְבֵּל אֶת הַמֹּחַ", עָנָה מָרְדְּכַי.
הִבַּטְתִּי בְּיָאִיר הַנּוֹשֵׁךְ אֶת שְׂפָתָיו, וְאָז רָאִיתִי בְּמוֹ עֵינַי אֶת הַנֵּר מַגִּיעַ מַמָּשׁ לְיָדָיו שֶׁל יָאִיר. הוּא פָּלַט זְעָקָה אֲיֻמָּה, וְהַנֵּר עָף מִיָּדָיו…
…הַיְשֵׁר לַעֲרֵמַת הַקַּשׁ!!!
וְזוֹ נִדְלְקָה בִּשְׁנִיָּה.
"מֶה עָשִׂיתָ?" צָעַק מָרְדְּכַי וְהֵנִיף יָדוֹ לְהַכּוֹת אֶת יָאִיר, אֲבָל יָאִיר נֶעֱלַם. הוּא בָּרַח.
גַּם אֲנִי בָּרַחְתִּי בִּמְהִירוּת מֵהַמִּקְלָט.
בַּחוּץ מָצָאתִי אֶת יָאִיר מַחְזִיק בְּיָדוֹ וּמְמָרֵר בִּבְכִי.
"מַה קָּרָה לְךָ?" שָׁאַלְתִּי.
"נִכְוֵיתִי", אָמַר וְהֶרְאָה אֶת יָדוֹ. אָכֵן הָיְתָה שָׁם כְּוִיָּה לֹא קְטַנָּה.
"יֵשׁ כָּאן בֶּרֶז", אָמַרְתִּי. "שִׂים אֶת הַיָּד תַּחַת הַבֶּרֶז בְּמֶשֶׁךְ רֶבַע שָׁעָה. כָּל עוֹד הַמַּיִם יִשְׁטְפוּ אֶת הַכְּוִיָּה לֹא יִכְאַב לְךָ, וְאַחֲרֵי רֶבַע שָׁעָה זֶה יַעֲבֹר".
"אֲבָל אֲנִי מְפַחֵד שֶׁמָּרְדְּכַי יַרְבִּיץ לִי", אָמַר.
"רָאִיתָ אֶת מָרְדְּכַי?" שָׁאַלְתִּי.
"לֹא, הוּא בַּמִּקְלָט", אָמַר יָאִיר.
הִבַּטְנוּ שְׁנֵינוּ בַּמִּקְלָט, וְהֵבַנּוּ מַה מַּשְׁמָעוּת הַדָּבָר.
עַנְנֵי עָשָׁן יָצְאוּ מִשָּׁם. מָרְדְּכַי נוֹתַר בְּתוֹךְ הַמִּקְלָט.
"אִמָּאלֶ'ה!" אָמַר יָאִיר, שֶׁאַף עַל פִּי שֶׁהָיָה קָטָן הֵבִין הֵיטֵב אֶת הַמַּשְׁמָעוּת.
"לֵךְ שְׁטֹף אֶת הַיָּד", אָמַרְתִּי, "אֲנִי אֲנַסֶּה לְהַצִּיל אוֹתוֹ".
יָאִיר רָץ לַבֶּרֶז, וַאֲנִי רַצְתִּי אֶל תּוֹךְ הַמִּקְלָט.
לִפְנֵי שֶׁנִּכְנַסְתִּי, נָשַׁמְתִּי נְשִׁימָה אֲרֻכָּה וְסָגַרְתִּי אֶת פִּי. יָדַעְתִּי שֶׁשְּׁאִיפַת הֶעָשָׁן מְסֻכֶּנֶת מִכֹּל. הֶעָשָׁן חוֹנֵק וְהוֹרֵס אֶת הָרֵאוֹת.
נִכְנַסְתִּי, סָקַרְתִּי אֶת הַמִּקְלָט וְרָאִיתִי אֶת מָרְדְּכַי שׁוֹכֵב עַל הָרִצְפָּה בֵּין הַלֶּהָבוֹת.
תָּפַסְתִּי בְּרַגְלָיו וְהִתְחַלְתִּי לִגְרֹר אוֹתוֹ הַחוּצָה. הוּא הָיָה דֵּי כָּבֵד, אֲבָל אֲנִי גָּרַרְתִּי אוֹתוֹ בְּרַגְלָיו כַּאֲחוּז טֵרוּף עַד שֶׁהִגַּעְתִּי לְפֶתַח הַמִּקְלָט, וְאָז הִנַּחְתִּי אוֹתוֹ. רַצְתִּי לְמַעְלָה, נִשְׁכַּבְתִּי עַל הַמַּדְרֵגוֹת בַּאֲפִיסַת כֹּחוֹת וְנָשַׁמְתִּי אֲוִיר נָקִי מְלֹא רֵאוֹתַי.
כְּשֶׁהִתְאוֹשַׁשְׁתִּי, רָאִיתִי כַּמָּה יְלָדִים עוֹמְדִים לְיָדִי.
הָיוּ אֵלֶּה "הַשּׁוֹלְטִים".
"מָה אַתָּה עוֹשֶׂה כָּאן?" שָׁאֲלוּ.
רָצִיתִי לוֹמַר לָהֶם שֶׁהִצַּלְתִּי אֶת מָרְדְּכַי.
"אֲנִי… אֲנִי… הִצַּלְתִּי… אֶת…"
וְאָז רָאִיתִי שֶׁמָּרְדְּכַי עוֹמֵד מֵאֲחוֹרֵיהֶם בָּרִיא וְשָׁלֵם, בְּלִי כְּוִיּוֹת וּבְלִי כְּלוּם.
הִבַּטְתִּי לְאָחוֹר, וּפִתְאוֹם הֵבַנְתִּי אֶת מִי מָשַׁכְתִּי.
מָשַׁכְתִּי אֶת הֶהָמָן הַגָּדוֹל וְהַכָּבֵד. הוּא שָׁכַב שָׁם בְּפֶתַח הַמִּקְלָט לְלֹא תְּנוּעָה, רֹאשׁוֹ שָׂרוּף מְעַט, אֲבָל בְּאֹפֶן כְּלָלִי שָׁלֵם לְגַמְרֵי עַל בְּגָדָיו וְהַקְּרָשִׁים שֶׁבְּתוֹכוֹ.
"הִצַּלְתָּ אֶת הָמָן, הָא?" אָמַר מְנַחֵם, אַחַד הַיְלָדִים, וְכֻלָּם צָחֲקוּ: "תַּגִּיד, אַתָּה הִשְׁתַּגַּעְתָּ? כִּמְעַט נִשְׂרַפְתָּ בְּעַצְמְךָ".
לֹא אָמַרְתִּי דָּבָר. רָצִיתִי לוֹמַר שֶׁחָשַׁבְתִּי שֶׁזֶּה מָרְדְּכַי, אֲבָל הֵבַנְתִּי שֶׁיִּהְיֶה זֶה טִפְּשִׁי מִצִּדִּי לוֹמַר זֹאת. הֵם פָּשׁוּט יִצְחֲקוּ לִי וִיבַזּוּ אוֹתִי.
"אַתָּה רִגַּלְתָּ אַחֲרֵינוּ?" שָׁאַל מְנַחֵם.
"לֹא, פָּשׁוּט עָבַרְתִּי כָּאן וְרָאִיתִי… רָאִיתִי עָשָׁן".
"נָנִיחַ", אָמַר מְנַחֵם, "אֲבָל אַל תָּבוֹא לְכָאן יוֹתֵר".
"וְאִם תְּסַפֵּר לְמִישֶׁהוּ אֵיפֹה הַמַּחֲבוֹא שֶׁלָּנוּ, אוֹי וַאֲבוֹי לְךָ", אָמַר טוֹבִיָּה.
"אָמְרוּ לְךָ פַּעַם שֶׁאַתָּה לֹא נוֹרְמָלִי?" שָׁאַל מְנַחֵם, "אַתָּה הָיִיתָ יָכוֹל לְהִשָּׂרֵף כָּאן בִּשְׁבִיל הֶהָמָן הַזֶּה".
הִבַּטְתִּי בּוֹ וְלֹא אָמַרְתִּי דָּבָר. יָדַעְתִּי שֶׁאִם אֹמַר אֶת הָאֱמֶת, הֵם יִפְּלוּ עַל הָרִצְפָּה מֵרֹב צְחוֹק. לֹא רָצִיתִי לָתֵת לָהֶם סִבָּה לִלְעֹג לִי.
מָרְדְּכַי לֹא אָמַר דָּבָר. הוּא נִרְאָה הָמוּם לְגַמְרֵי.
הָלַכְתִּי הַבַּיְתָה וְהִרְגַּשְׁתִּי נוֹרָא. לֹא רַק שֶׁסִּכַּנְתִּי אֶת עַצְמִי לְהַצִּיל יֶלֶד, וְהַיֶּלֶד הַזֶּה בִּכְלָל לֹא הָיָה שָׁם, אֶלָּא הִצַּלְתִּי לָהֶם אֶת הָמָן, שֶׁלֹּא הָיִיתִי שֻׁתָּף לוֹ כְּלָל, וְהֵם אֲפִלּוּ אֵינָם מַעֲרִיכִים אוֹתִי.
הִגַּעְתִּי לְבֵיתִי, נִכְנַסְתִּי לְחַדְרִי וְנִשְׁכַּבְתִּי עַל הַמִּטָּה בּוֹכֶה וּמִתְיַפֵּחַ.
אִמִּי נִכְנְסָה לַחֶדֶר וְשָׁאֲלָה מַה קָּרָה. לֹא יָכֹלְתִּי אֲפִלּוּ לַעֲנוֹת לָהּ.
"אֲנִי מַרְגִּישָׁה רֵיחַ שֶׁל עָשָׁן", אָמְרָה אִמִּי. "תַּגִּיד מַה קָּרָה".
רָצִיתִי לְסַפֵּר לָהּ הַכֹּל, אֲבָל לֹא מָצָאתִי אֶת הַמִּלִּים. כְּבָר אָמַרְתִּי לָכֶם שֶׁזּוֹ הַבְּעָיָה שֶׁלִּי.
נִרְדַּמְתִּי, בְּלִי לְהִתְקַלֵּחַ וּבְלִי לְהַחְלִיף בְּגָדִים.
לְמָחֳרָת הָלַכְתִּי לַכִּתָּה בְּלִי שׁוּם כֹּחַ. רָאִיתִי אֶת הַיְלָדִים שֶׁפָּגַשְׁתִּי אֶתְמוֹל. הֵם הִבִּיטוּ בִּי בְּמַבָּט מוּזָר. קָלַטְתִּי שֶׁהֵם לֹא יְסַפְּרוּ כְּלוּם עַל מַה שֶּׁקָּרָה, כִּי הֵם פּוֹחֲדִים שֶׁאֲסַפֵּר אֵיפֹה מְקוֹם הַמִּסְתּוֹר שֶׁל הֶהָמָן שֶׁלָהֶם. הָיְתָה בֵּינֵינוּ מֵעֵין הַסְכָּמָה שֶׁבִּשְׁתִיקָה: הֵם לֹא יִלְעֲגוּ לִי, וַאֲנִי לֹא אֲסַפֵּר אֶת סוֹדָם. זֶה הָיָה נוֹחַ לִי וְהָיָה נוֹחַ לָהֶם.
יוֹמַיִם לְאַחַר מִכֵּן הָיָה לַ"ג בָּעֹמֶר.
הִגַּעְתִּי לַמְּדוּרָה בְּלִי חֵשֶׁק. לֹא רָצִיתִי לְהִשְׁתַּתֵּף בָּהּ, אֲבָל יָדַעְתִּי שֶׁאִם אֵעָדֵר, אַעֲנִישׁ אֶת עַצְמִי. לֹא רָאִיתִי כָּל טַעַם בְּכָךְ.
הִגַּעְתִּי לַאֲתַר הַמְּדוּרָה. הִיא הָיְתָה גְּבוֹהָה מְאֹד, וּבְרֹאשָׁהּ הִתְנוֹסֵס הֶהָמָן הָעֲנָק שֶׁהִצַּלְתִּי.
הוּא הָיָה מַרְשִׁים בְּיוֹתֵר. גָּבְהוֹ יוֹתֵר מִגֹּבַהּ אָדָם, וְהוּא הָיָה שָׁמֵן מְאֹד. הוּא בָּלַט לְמֵרָחוֹק.
הָעֶרֶב יָרַד, וְיַלְדֵי הַשְּׁכוּנָה הִמְתִּינוּ לַהַדְלָקָה.
"מִי יַדְלִיק אֶת הַמְּדוּרָה?" שָׁאֲלוּ כֻּלָּם.
כֻּלָּם הִמְתִּינוּ לְמָרְדְּכַי. הוּא זֶה שֶׁמַּדְלִיק בְּדֶרֶךְ כְּלָל.
מָרְדְּכַי לָקַח אֶת הָאֲבוּקָה, הִדְלִיק אוֹתָהּ, וְאָז עָצַר לְפֶתַע וְאָמַר: "אֲנִי רוֹצֶה לְסַפֵּר לָכֶם מַשֶּׁהוּ".
כֻּלָּם הִשְׁתַּתְּקוּ.
"אַתֶּם רוֹאִים אֶת הֶהָמָן הָעֲנָק הַזֶּה?" שָׁאַל, וְכֻלָּם הִנְהֲנוּ. מִי לֹא רָאָה?
"לִפְנֵי כַּמָּה יָמִים הֶהָמָן כִּמְעַט נִשְׂרַף. מָקוֹם הַמַּחֲבוֹא שֶׁלּוֹ עָלָה בָּאֵשׁ, וּמִישֶׁהוּ פָּשׁוּט גָּרַר אוֹתוֹ מִשָּׁם וְהִצִּיל אוֹתוֹ תּוֹךְ כְּדֵי סִכּוּן חַיָּיו".
רַחַשׁ עָבַר בַּקָּהָל.
כָּל הַחֲבֵרִים שֶׁהָיוּ שֻׁתָּפִים לַסּוֹד לֹא יָדְעוּ אִם לְהַעֲרִיךְ אֶת הַיֶּלֶד שֶׁהִצִּיל אֶת הָמָן אוֹ לִלְעֹג לוֹ. מִצַּד אֶחָד עָבְדוּ עַל הֶהָמָן הַזֶּה הֲמוֹן זְמַן וְהִשְׁקִיעוּ בּוֹ הֲמוֹן כֹּחוֹת, מִצַּד שֵׁנִי זֶה מַמָּשׁ מְטֹרָף לְסַכֵּן אֶת הַחַיִּים בִּשְׁבִיל לְהַצִּיל אֵיזוֹ בֻּבָּה שֶׁל סְמַרְטוּטִים.
"אֲבָל יֵשׁ מַשֶּׁהוּ שֶׁאֲפִלּוּ הַחֲבֵרִים שֶׁלִּי הַקְּרוֹבִים לֹא יוֹדְעִים. הַיֶּלֶד שֶׁנִּכְנַס לַמִּקְלָט הַבּוֹעֵר לֹא נִכְנַס כְּדֵי לְהַצִּיל אֶת הָמָן. הוּא נִכְנַס לְהַצִּיל אֶת מָרְדְּכַי. כֵּן, אֲנִי הָיִיתִי שָׁם עִם אָחִי יָאִיר, וְהַנֵּר שֶׁהֶחְזִיק בְּיָדוֹ גָּרַם לַשְּׂרֵפָה, הַרְבֵּה בְּאַשְׁמָתִי. לֹא הִקְשַׁבְתִּי לוֹ כְּשֶׁאָמַר שֶׁהַנֵּר נִגְמַר, וּכְשֶׁהַנֵּר הִגִּיעַ לַיָּד שֶׁלּוֹ הוּא שָׁמַט אוֹתוֹ וְכָל הַקַּשׁ נִדְלַק.
יָאִיר בָּרַח מֵהַכְּנִיסָה לַמִּקְלָט וַאֲנִי נִמְלַטְתִּי לְתוֹךְ הַמִּקְלָט, וְשָׁם מָצָאתִי יְצִיאָה דֶּרֶךְ אֲרֻבַּת הָאִוְרוּר.
הָיָה שָׁם יֶלֶד נוֹסָף, וְהוּא טִפֵּל בְּאָחִי יָאִיר. כְּשֶׁזֶּה אָמַר לוֹ שֶׁאֲנִי נִשְׁאַרְתִּי בְּתוֹךְ הַמִּקְלָט, הוּא פָּשׁוּט רָץ לְתוֹכוֹ כְּדֵי לְהַצִּיל אוֹתִי.
הוּא רָאָה בַּחֹשֶׁךְ דְּמוּת שֶׁל אָדָם עִם הַנַּעֲלַיִם הַיְשָׁנוֹת שֶׁלִּי, וּמָשַׁךְ אוֹתָהּ בָּרַגְלַיִם. רַק כְּשֶׁיָּצָא רָאָה אוֹתִי וְהֵבִין שֶׁהִצִּיל אֶת הָמָן בִּמְקוֹם אֶת מָרְדְּכַי. אֲבָל רַק אֲנִי וְהוּא יָדַעְנוּ מֶה הָיְתָה כַּוָּנָתוֹ.
אֲנִי חוֹשֵׁב עַל כָּךְ כְּבָר כַּמָּה יָמִים", הוֹסִיף מָרְדְּכַי, "וַאֲנִי מֵבִין יוֹתֵר וְיוֹתֵר שֶׁיֵּשׁ לָנוּ בַּכִּתָּה יֶלֶד גִּבּוֹר אֲמִתִּי. הַיֶּלֶד הַזֶּה לֹא חָבֵר שֶׁלִּי, הִתְיַחַסְתִּי אֵלָיו לֹא יָפֶה, וּבְכָל זֹאת הוּא הִסְכִּים לְסַכֵּן אֶת חַיָּיו כְּדֵי לְהַצִּיל אוֹתִי. הֶחְלַטְתִּי שֶׁאֲנִי חַיָּב לְסַפֵּר אֶת זֶה לְכָל הַיְלָדִים, שֶׁיַּעֲרִיכוּ אוֹתוֹ, וְגַם לְכַבֵּד אוֹתוֹ בְּהַדְלָקַת הַמְּדוּרָה".
כֻּלָּם הִבִּיטוּ זֶה בָּזֶה, מְסֻקְרָנִים לָדַעַת מִי הוּא הַגִּבּוֹר.
"אֱלִיעֶזֶר, בּוֹא תַּדְלִיק אֶת הַמְּדוּרָה", אָמַר מָרְדְּכַי.
כֻּלָּם הִבִּיטוּ בִּי בְּתַדְהֵמָה: "מָה? הוּא?" אָמְרוּ כֻּלָּם, וְאָז הֵחֵלּוּ לִמְחֹא כַּף.
נָטַלְתִּי בְּיָדַיִם רוֹעֲדוֹת אֶת הָאֲבוּקָה, נִגַּשְׁתִּי אֶל הַמְּדוּרָה וְהִדְלַקְתִּי אוֹתָהּ לְקוֹל תְּשׁוּאוֹת יַלְדֵי הַשְּׁכוּנָה וְהַכִּתָּה.
תִּימְרוֹת הָאֵשׁ הֵחֵלּוּ לֶאֱחֹז בְּהָמָן.
וַאֲנִי יָדַעְתִּי שֶׁבָּרְגָעִים שֶׁבָּהֶם יִשָּׂרֵף הֶהָמָן שֶׁהִצַּלְתִּי, הֶהָמָן הַפְּרָטִי שֶׁלִּי יַהֲפֹךְ בַּחֲזָרָה לְמָרְדְּכַי.
